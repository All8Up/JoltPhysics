100

102  node->mNodeProperties[i] = this_node->GetTriangleCount() << TRIANGLE_COUNT_SHIFT;
103  if (this_node->GetTriangleCount() >= TRIANGLE_COUNT_MASK)
104  {
105  outError = "NodeCodecQuadTreeHalfFloat: Too many triangles";
106  return uint(-1);
107  }
108  }
109  else
110  {
111  // Make this an invalid triangle node
112  node->mNodeProperties[i] = uint32(TRIANGLE_COUNT_MASK) << TRIANGLE_COUNT_SHIFT;
113 
114  // Make bounding box invalid
115  node->mBoundsMinX[i] = HALF_FLT_MAX;
116  node->mBoundsMinY[i] = HALF_FLT_MAX;
117  node->mBoundsMinZ[i] = HALF_FLT_MAX;
118  node->mBoundsMaxX[i] = HALF_FLT_MAX;
119  node->mBoundsMaxY[i] = HALF_FLT_MAX;
120  node->mBoundsMaxZ[i] = HALF_FLT_MAX;
121  }

125  for (int i = 0; i < NumChildrenPerNode; ++i)
126  {

128  outChildBoundsMax[i] = inNodeBoundsMax;

135  bool NodeFinalize(const AABBTreeBuilder::Node *inNode, uint inNodeStart, uint inTrianglesStart, uint inNumChildren, const uint *inChildrenNodeStart, const uint *inChildrenTrianglesStart, ByteBuffer &ioBuffer, string &outError) const
136  {
137  if (!inNode->HasChildren())
138  return true;
139 
140  Node *node = ioBuffer.Get<Node>(inNodeStart);
141  for (uint i = 0; i < inNumChildren; ++i)
142  {
143  // If there are triangles, use the triangle offset otherwise use the node offset
144  uint offset = node->mNodeProperties[i] != 0? inChildrenTrianglesStart[i] : inChildrenNodeStart[i];
145  if (offset & OFFSET_NON_SIGNIFICANT_MASK)
146  {
147  outError = "NodeCodecQuadTreeHalfFloat: Internal Error: Offset has non-signifiant bits set";
148  return false;
149  }
150  offset >>= OFFSET_NON_SIGNIFICANT_BITS;
151  if (offset & ~OFFSET_MASK)
152  {
153  outError = "NodeCodecQuadTreeHalfFloat: Offset too large. Too much data.";
154  return false;
155  }
156 
157  // Store offset of next node / triangles
158  node->mNodeProperties[i] |= offset;
159  }
160 
161  return true;
162  }

165  bool Finalize(Header *outHeader, const AABBTreeBuilder::Node *inRoot, uint inRootNodeStart, uint inRootTrianglesStart, string &outError) const
166  {
167  uint offset = inRoot->HasChildren()? inRootNodeStart : inRootTrianglesStart;
168  if (offset & OFFSET_NON_SIGNIFICANT_MASK)
169  {
170  outError = "NodeCodecQuadTreeHalfFloat: Internal Error: Offset has non-signifiant bits set";
171  return false;
172  }
173  offset >>= OFFSET_NON_SIGNIFICANT_BITS;
174  if (offset & ~OFFSET_MASK)
175  {
176  outError = "NodeCodecQuadTreeHalfFloat: Offset too large. Too much data.";
177  return false;
178  }
179 
180  inRoot->mBounds.mMin.StoreFloat3(&outHeader->mRootBoundsMin);
181  inRoot->mBounds.mMax.StoreFloat3(&outHeader->mRootBoundsMax);
182  outHeader->mRootProperties = offset + (inRoot->GetTriangleCount() << TRIANGLE_COUNT_SHIFT);
183  if (inRoot->GetTriangleCount() >= TRIANGLE_COUNT_MASK)
184  {
185  outError = "NodeCodecQuadTreeHalfFloat: Too many triangles";
186  return false;
187  }
188 
189  return true;
190  }

219  template <class TriangleContext, class Visitor>
220  inline void WalkTree(const uint8 *inBufferStart, const TriangleContext &inTriangleContext, Visitor &ioVisitor)
221  {
222  do
223  {
224  // Test if node contains triangles
225  uint32 node_properties = mNodeStack[mTop];
226  uint32 tri_count = node_properties >> TRIANGLE_COUNT_SHIFT;
227  if (tri_count == 0)
228  {
229  const Node *node = reinterpret_cast<const Node *>(inBufferStart + (node_properties << OFFSET_NON_SIGNIFICANT_BITS));
230 
231  // Unpack bounds
232  UVec4 bounds_minxy = UVec4::sLoadInt4(reinterpret_cast<const uint32 *>(&node->mBoundsMinX[0]));
233  Vec4 bounds_minx = HalfFloatConversion::ToFloat(bounds_minxy);
234  Vec4 bounds_miny = HalfFloatConversion::ToFloat(bounds_minxy.Swizzle<SWIZZLE_Z, SWIZZLE_W, SWIZZLE_UNUSED, SWIZZLE_UNUSED>());
235 
236  UVec4 bounds_minzmaxx = UVec4::sLoadInt4(reinterpret_cast<const uint32 *>(&node->mBoundsMinZ[0]));
237  Vec4 bounds_minz = HalfFloatConversion::ToFloat(bounds_minzmaxx);
238  Vec4 bounds_maxx = HalfFloatConversion::ToFloat(bounds_minzmaxx.Swizzle<SWIZZLE_Z, SWIZZLE_W, SWIZZLE_UNUSED, SWIZZLE_UNUSED>());
239 
240  UVec4 bounds_maxyz = UVec4::sLoadInt4(reinterpret_cast<const uint32 *>(&node->mBoundsMaxY[0]));
241  Vec4 bounds_maxy = HalfFloatConversion::ToFloat(bounds_maxyz);
242  Vec4 bounds_maxz = HalfFloatConversion::ToFloat(bounds_maxyz.Swizzle<SWIZZLE_Z, SWIZZLE_W, SWIZZLE_UNUSED, SWIZZLE_UNUSED>());
243 
244  // Load properties for 4 children
245  UVec4 properties = UVec4::sLoadInt4(&node->mNodeProperties[0]);
246 
247  // Check which sub nodes to visit
248  int num_results = ioVisitor.VisitNodes(bounds_minx, bounds_miny, bounds_minz, bounds_maxx, bounds_maxy, bounds_maxz, properties, mTop);
249 
250  // Push them onto the stack
251  JPH_ASSERT(mTop + 4 < StackSize);
252  properties.StoreInt4(&mNodeStack[mTop]);
253  mTop += num_results;
254  }
255  else if (tri_count != TRIANGLE_COUNT_MASK) // TRIANGLE_COUNT_MASK indicates a padding node, normally we shouldn't visit these nodes but when querying with a big enough box you could touch HALF_FLT_MAX (about 65K)
256  {
257  // Node contains triangles, do individual tests
258  uint32 triangle_block_id = node_properties & OFFSET_MASK;
259  const void *triangles = sGetTriangleBlockStart(inBufferStart, triangle_block_id);
260 
261  ioVisitor.VisitTriangles(inTriangleContext, mRootBoundsMin, mRootBoundsMax, triangles, tri_count, triangle_block_id);
262  }
263 
264  // Check if we're done
265  if (ioVisitor.ShouldAbort())
266  break;
267 
268  // Fetch next node until we find one that the visitor wants to see
269  do
270  --mTop;
271  while (mTop >= 0 && !ioVisitor.ShouldVisitNode(mTop));
272  }
273  while (mTop >= 0);
274  }
275 
277  bool IsDoneWalking() const
278  {
279  return mTop < 0;
280  }
281 
282  private:
283  Vec3 mRootBoundsMin;
284  Vec3 mRootBoundsMax;
285  uint32 mNodeStack[StackSize];
286  int mTop = 0;
287  };
