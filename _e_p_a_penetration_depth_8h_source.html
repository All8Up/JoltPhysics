95  template <typename AE, typename BE>
96  EStatus GetPenetrationDepthStepGJK(const AE &inAExcludingConvexRadius, float inConvexRadiusA, const BE &inBExcludingConvexRadius, float inConvexRadiusB, float inTolerance, Vec3 &ioV, Vec3 &outPointA, Vec3 &outPointB)
97  {
98  JPH_PROFILE_FUNCTION();
99 
100  // Don't supply a zero ioV, we only want to get points on the hull of the Minkowsky sum and not internal points
101  JPH_ASSERT(!ioV.IsNearZero());
102 
103  // Get closest points
104  float combined_radius = inConvexRadiusA + inConvexRadiusB;
105  float combined_radius_sq = combined_radius * combined_radius;
106  float closest_points_dist_sq = mGJK.GetClosestPoints(inAExcludingConvexRadius, inBExcludingConvexRadius, inTolerance, combined_radius_sq, ioV, outPointA, outPointB);
107  if (closest_points_dist_sq > combined_radius_sq)
108  {
109  // No collision
110  return EStatus::NotColliding;
111  }
112  if (closest_points_dist_sq > 0.0f)
113  {
114  // Collision within convex radius, adjust points for convex radius
115  float v_len = sqrt(closest_points_dist_sq); // GetClosestPoints function returns |ioV|^2 when return value < FLT_MAX
116  outPointA += ioV * (inConvexRadiusA / v_len);
117  outPointB -= ioV * (inConvexRadiusB / v_len);
118  return EStatus::Colliding;
119  }
120 
121  return EStatus::Indeterminate;
122  }

136  template <typename AI, typename BI>
137  bool GetPenetrationDepthStepEPA(const AI &inAIncludingConvexRadius, const BI &inBIncludingConvexRadius, float inTolerance, Vec3 &outV, Vec3 &outPointA, Vec3 &outPointB)
138  {
139  JPH_PROFILE_FUNCTION();
140 
141  // Check that the tolerance makes sense (smaller value than this will just result in needless iterations)
142  JPH_ASSERT(inTolerance >= FLT_EPSILON);
143 
144  // Fetch the simplex from GJK algorithm
145  SupportPoints support_points;
146  mGJK.GetClosestPointsSimplex(support_points.mY.data(), support_points.mP, support_points.mQ, support_points.mY.GetSizeRef());
147 
148  // Fill up the amount of support points to 4
149  switch (support_points.mY.size())
150  {
151  case 1:
152  {
153  // 1 vertex, which must be at the origin, which is useless for our purpose
154  JPH_ASSERT(support_points.mY[0].IsNearZero(1.0e-8f));
155  support_points.mY.pop_back();
156 
157  // Add support points in 4 directions to form a tetrahedron around the origin
158  int p1, p2, p3, p4;
159  (void)support_points.Add(inAIncludingConvexRadius, inBIncludingConvexRadius, Vec3(0, 1, 0), p1);
160  (void)support_points.Add(inAIncludingConvexRadius, inBIncludingConvexRadius, Vec3(-1, -1, -1), p2);
161  (void)support_points.Add(inAIncludingConvexRadius, inBIncludingConvexRadius, Vec3(1, -1, -1), p3);
162  (void)support_points.Add(inAIncludingConvexRadius, inBIncludingConvexRadius, Vec3(0, -1, 1), p4);
163  JPH_ASSERT(p1 == 0);
164  JPH_ASSERT(p2 == 1);
165  JPH_ASSERT(p3 == 2);
166  JPH_ASSERT(p4 == 3);
167  break;
168  }
169 
170  case 2:
171  {
172  // Two vertices, create 3 extra by taking perpendicular axis and rotating it around in 120 degree increments
173  Vec3 axis = (support_points.mY[1] - support_points.mY[0]).Normalized();
174  Mat44 rotation = Mat44::sRotation(axis, DegreesToRadians(120.0f));
175  Vec3 dir1 = axis.GetNormalizedPerpendicular();
176  Vec3 dir2 = rotation * dir1;
177  Vec3 dir3 = rotation * dir2;
178  int p1, p2, p3;
179  (void)support_points.Add(inAIncludingConvexRadius, inBIncludingConvexRadius, dir1, p1);
180  (void)support_points.Add(inAIncludingConvexRadius, inBIncludingConvexRadius, dir2, p2);
181  (void)support_points.Add(inAIncludingConvexRadius, inBIncludingConvexRadius, dir3, p3);
182  JPH_ASSERT(p1 == 2);
183  JPH_ASSERT(p2 == 3);
184  JPH_ASSERT(p3 == 4);
185  break;
186  }
187 
188  case 3:
189  case 4:
190  // We already have enough points
191  break;
192  }
193 
194  // Create hull out of the initial points
195  JPH_ASSERT(support_points.mY.size() >= 3);
196  EPAConvexHullBuilder hull(support_points.mY);
197  hull.Initialize(0, 1, 2);
198  for (typename Points::size_type i = 3; i < support_points.mY.size(); ++i)
199  {
200  float dist_sq;
201  Triangle *t = hull.FindFacingTriangle(support_points.mY[i], dist_sq);
202  if (t != nullptr)
203  {
204  EPAConvexHullBuilder::NewTriangles new_triangles;
205  if (!hull.AddPoint(t, i, FLT_MAX, new_triangles))
206  {
207  // We can't recover from a failure to add a point to the hull because the old triangles have been unlinked already.
208  // Assume no collision. This can happen if the shapes touch in 1 point (or plane) in which case the hull is degenerate.
209  return false;
210  }
211  }
212  }
213 
214  // Loop until we are sure that the origin is inside the hull
215  for (;;)
216  {
217  // Get the next closest triangle
218  Triangle *t = hull.PeekClosestTriangleInQueue();
219 
220  // Don't process removed triangles, just free them (because they're in a heap we don't remove them earlier since we would have to rebuild the sorted heap)
221  if (t->mRemoved)
222  {
223  hull.PopClosestTriangleFromQueue();
224 
225  // If we run out of triangles, we couldn't include the origin in the hull so there must be very little penetration and we report no collision.
226  if (!hull.HasNextTriangle())
227  return false;
228 
229  hull.FreeTriangle(t);
230  continue;
231  }
232 
233  // If the closest to the triangle is zero or positive, the origin is in the hull and we can proceed to the main algorithm
234  if (t->mClosestLenSq >= 0.0f)
235  break;
236 
237  // Remove the triangle from the queue before we start adding new ones (which may result in a new closest triangle at the front of the queue)
238  hull.PopClosestTriangleFromQueue();
239 
240  // Add a support point to get the origin inside the hull
241  int new_index;
242  Vec3 w = support_points.Add(inAIncludingConvexRadius, inBIncludingConvexRadius, t->mNormal, new_index);
243 
244 #ifdef JPH_EPA_CONVEX_BUILDER_DRAW
245  // Draw the point that we're adding
246  hull.DrawMarker(w, Color::sRed, 1.0f);
247  hull.DrawWireTriangle(*t, Color::sRed);
248  hull.DrawState();
249 #endif
250 
251  // Add the point to the hull, if we fail we terminate and report no collision
252  EPAConvexHullBuilder::NewTriangles new_triangles;
253  if (!t->IsFacing(w) || !hull.AddPoint(t, new_index, FLT_MAX, new_triangles))
254  return false;
255 
256  // If the triangle was removed we can free it now
257  if (t->mRemoved)
258  hull.FreeTriangle(t);
259 
260  // If we run out of triangles or points, we couldn't include the origin in the hull so there must be very little penetration and we report no collision.
261  if (!hull.HasNextTriangle() || support_points.mY.size() >= cMaxPointsToIncludeOriginInHull)
262  return false;
263  }
264 
265  // Current closest distance to origin
266  float closest_dist_sq = FLT_MAX;
267 
268  // Remember last good triangle
269  Triangle *last = nullptr;
270 
271  // Loop until closest point found
272  do
273  {
274  // Get closest triangle to the origin
275  Triangle *t = hull.PopClosestTriangleFromQueue();
276 
277  // Don't process removed triangles, just free them (because they're in a heap we don't remove them earlier since we would have to rebuild the sorted heap)
278  if (t->mRemoved)
279  {
280  hull.FreeTriangle(t);
281  continue;
282  }
283 
284  // Check if next triangle is further away than closest point, we've found the closest point
285  if (t->mClosestLenSq >= closest_dist_sq)
286  break;
287 
288  // Replace last good with this triangle
289  if (last != nullptr)
290  hull.FreeTriangle(last);
291  last = t;
292 
293  // Add support point in direction of normal of the plane
294  // Note that the article uses the closest point between the origin and plane, but this always has the exact same direction as the normal (if the origin is behind the plane)
295  // and this way we do less calculations and lose less precision
296  int new_index;
297  Vec3 w = support_points.Add(inAIncludingConvexRadius, inBIncludingConvexRadius, t->mNormal, new_index);
298 
299  // Project w onto the triangle normal
300  float dot = t->mNormal.Dot(w);
301 
302  // Check if we just found a separating axis. This can happen if the shape shrunk by convex radius and then expanded by
303  // convex radius is bigger then the original shape due to inaccuracies in the shrinking process.
304  if (dot < 0.0f)
305  return false;
306 
307  // Get the distance squared (along normal) to the support point
308  float dist_sq = dot * dot / t->mNormal.LengthSq();
309 
310 #ifdef JPH_EPA_CONVEX_BUILDER_DRAW
311  // Draw the point that we're adding
312  hull.DrawMarker(w, Color::sPurple, 1.0f);
313  hull.DrawWireTriangle(*t, Color::sPurple);
314  hull.DrawState();
315 #endif
316 
317  // If the error became small enough, we've converged
318  if (dist_sq - t->mClosestLenSq < t->mClosestLenSq * inTolerance)
319  break;
320 
321  // Keep track of the minimum distance
322  closest_dist_sq = min(closest_dist_sq, dist_sq);
323 
324  // If the triangle thinks this point is not front facing, we've reached numerical precision and we're done
325  if (!t->IsFacing(w))
326  break;
327 
328  // Add point to hull
329  EPAConvexHullBuilder::NewTriangles new_triangles;
330  if (!hull.AddPoint(t, new_index, closest_dist_sq, new_triangles))
331  break;
332 
333  // If the hull is starting to form defects then we're reaching numerical precision and we have to stop
334  bool has_defect = false;
335  for (Triangle *nt : new_triangles)
336  if (nt->IsFacingOrigin())
337  {
338  has_defect = true;
339  break;
340  }
341  if (has_defect)
342  break;
343  }
344  while (hull.HasNextTriangle() && support_points.mY.size() < cMaxPoints);
345 
346  // Determine closest points, if last == null it means the hull was a plane so there's no penetration
347  if (last == nullptr)
348  return false;
349 
350  // Should be an interior point
351  JPH_ASSERT(last->mClosestPointInterior);
352 
353  // Calculate penetration by getting the vector from the origin to the closest point on the triangle:
354  // distance = (centroid - origin) . normal / |normal|, closest = origin + distance * normal / |normal|
355  outV = (last->mCentroid.Dot(last->mNormal) / last->mNormal.LengthSq()) * last->mNormal;
356 
357  // If penetration is near zero, treat this as a non collision since we cannot find a good normal
358  if (outV.IsNearZero())
359  return false;
360 
361  // Use the barycentric coordinates for the closest point to the origin to find the contact points on A and B
362  Vec3 p0 = support_points.mP[last->mEdge[0].mStartIdx];
363  Vec3 p1 = support_points.mP[last->mEdge[1].mStartIdx];
364  Vec3 p2 = support_points.mP[last->mEdge[2].mStartIdx];
365 
366  Vec3 q0 = support_points.mQ[last->mEdge[0].mStartIdx];
367  Vec3 q1 = support_points.mQ[last->mEdge[1].mStartIdx];
368  Vec3 q2 = support_points.mQ[last->mEdge[2].mStartIdx];
369 
370  if (last->mLambdaRelativeTo0)
371  {
372  // y0 was the reference vertex
373  outPointA = p0 + last->mLambda[0] * (p1 - p0) + last->mLambda[1] * (p2 - p0);
374  outPointB = q0 + last->mLambda[0] * (q1 - q0) + last->mLambda[1] * (q2 - q0);
375  }
376  else
377  {
378  // y1 was the reference vertex
379  outPointA = p1 + last->mLambda[0] * (p0 - p1) + last->mLambda[1] * (p2 - p1);
380  outPointB = q1 + last->mLambda[0] * (q0 - q1) + last->mLambda[1] * (q2 - q1);
381  }
382 
383  return true;
384  }

390  bool GetPenetrationDepth(const AE &inAExcludingConvexRadius, const AI &inAIncludingConvexRadius, float inConvexRadiusA, const BE &inBExcludingConvexRadius, const BI &inBIncludingConvexRadius, float inConvexRadiusB, float inCollisionToleranceSq, float inPenetrationTolerance, Vec3 &ioV, Vec3 &outPointA, Vec3 &outPointB)
391  {
392  // Check result of collision detection
393  switch (GetPenetrationDepthStepGJK(inAExcludingConvexRadius, inConvexRadiusA, inBExcludingConvexRadius, inConvexRadiusB, inCollisionToleranceSq, ioV, outPointA, outPointB))
394  {
395  case EPAPenetrationDepth::EStatus::Colliding:
396  return true;
397 
398  case EPAPenetrationDepth::EStatus::NotColliding:
399  return false;
400 
401  case EPAPenetrationDepth::EStatus::Indeterminate:
402  return GetPenetrationDepthStepEPA(inAIncludingConvexRadius, inBIncludingConvexRadius, inPenetrationTolerance, ioV, outPointA, outPointB);
403  }
404 
405  JPH_ASSERT(false);
406  return false;
407  }

427  bool CastShape(Mat44Arg inStart, Vec3Arg inDirection, float inCollisionTolerance, float inPenetrationTolerance, const A &inA, const B &inB, float inConvexRadiusA, float inConvexRadiusB, bool inReturnDeepestPoint, float &ioLambda, Vec3 &outPointA, Vec3 &outPointB, Vec3 &outContactNormal)
428  {
429  // First determine if there's a collision at all
430  if (!mGJK.CastShape(inStart, inDirection, inCollisionTolerance, inA, inB, inConvexRadiusA, inConvexRadiusB, ioLambda, outPointA, outPointB, outContactNormal))
431  return false;
432 
433  if (inReturnDeepestPoint
434  && ioLambda == 0.0f // Only when lambda = 0 we can have the bodies overlap
435  && (inConvexRadiusA + inConvexRadiusB == 0.0f // When no convex radius was provided we can never trust contact points at lambda = 0
436  || outContactNormal.LengthSq() <= inCollisionTolerance * inCollisionTolerance)) // When our contact normal is too small, we don't have an accurate result and need to run the EPA algorithm
437  {
438  // If we're initially intersecting, we need to run the EPA algorithm in order to find the deepest contact point

440  AddConvexRadius<B> add_convex_b(inB, inConvexRadiusB);
441  TransformedConvexObject<AddConvexRadius<A>> transformed_a(inStart, add_convex_a);
442  if (!GetPenetrationDepthStepEPA(transformed_a, add_convex_b, inPenetrationTolerance, outContactNormal, outPointA, outPointB))
443  return false;
444  }
445 
446  return true;
447  }
