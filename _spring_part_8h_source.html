21  inline void CalculateSpringProperties(float inDeltaTime, float inInvEffectiveMass, float inBias, float inC, float inFrequency, float inDamping, float &outEffectiveMass)
22  {
23  outEffectiveMass = 1.0f / inInvEffectiveMass;
24 
25  // Soft constraints as per: Soft Contraints: Reinventing The Spring - Erin Catto - GDC 2011
26  if (inFrequency > 0.0f)
27  {
28  // Calculate angular frequency
29  float omega = 2.0f * JPH_PI * inFrequency;
30 
31  // Calculate spring constant k and drag constant c (page 45)
32  float k = outEffectiveMass * Square(omega);
33  float c = 2.0f * outEffectiveMass * inDamping * omega;
34 
35  // Note that the calculation of beta and gamma below are based on the solution of an implicit Euler integration scheme
36  // This scheme is unconditionally stable but has built in damping, so even when you set the damping ratio to 0 there will still
37  // be damping. See page 16 and 32.
38 
39  // Calculate softness (gamma in the slides)
40  // See page 34 and note that the gamma needs to be divided by delta time since we're working with impulses rather than forces:
41  // softness = 1 / (dt * (c + dt * k))
42  mSoftness = 1.0f / (inDeltaTime * (c + inDeltaTime * k));
43 
44  // Calculate bias factor (baumgarte stabilization):
45  // beta = dt * k / (c + dt * k) = dt * k^2 * softness
46  // b = beta / dt * C = dt * k * softness * C;
47  mBias = inBias + inDeltaTime * k * mSoftness * inC;
48 
49  // Update the effective mass, see post by Erin Catto: http://www.bulletphysics.org/Bullet/phpBB3/viewtopic.php?f=4&t=1354
50  //
51  // Newton's Law:
52  // M * (v2 - v1) = J^T * lambda
53  //
54  // Velocity constraint with softness and Baumgarte:
55  // J * v2 + softness * lambda + b = 0
56  //
57  // where b = beta * C / dt
58  //
59  // We know everything except v2 and lambda.
60  //
61  // First solve Newton's law for v2 in terms of lambda:
62  //
63  // v2 = v1 + M^-1 * J^T * lambda
64  //
65  // Substitute this expression into the velocity constraint:
66  //
67  // J * (v1 + M^-1 * J^T * lambda) + softness * lambda + b = 0
68  //
69  // Now collect coefficients of lambda:
70  //
71  // (J * M^-1 * J^T + softness) * lambda = - J * v1 - b
72  //
73  // Now we define:
74  //
75  // K = J * M^-1 * J^T + softness
76  //
77  // So our new effective mass is K^-1
78  outEffectiveMass = 1.0f / (inInvEffectiveMass + mSoftness);
79  }
80  else
81  {
82  mSoftness = 0.0f;
83  mBias = inBias;
84  }
85  }
86 
88  inline bool IsActive() const
89  {
90  return mSoftness != 0.0f;
91  }
92 
94  inline float GetBias(float inTotalLambda) const
95  {
96  // Remainder of post by Erin Catto: http://www.bulletphysics.org/Bullet/phpBB3/viewtopic.php?f=4&t=1354
97  //
98  // Each iteration we are not computing the whole impulse, we are computing an increment to the impulse and we are updating the velocity.
99  // Also, as we solve each constraint we get a perfect v2, but then some other constraint will come along and mess it up.
100  // So we want to patch up the constraint while acknowledging the accumulated impulse and the damaged velocity.
101  // To help with that we use P for the accumulated impulse and lambda as the update. Mathematically we have:
102  //
103  // M * (v2new - v2damaged) = J^T * lambda
104  // J * v2new + softness * (total_lambda + lambda) + b = 0
105  //
106  // If we solve this we get:
107  //
108  // v2new = v2damaged + M^-1 * J^T * lambda
109  // J * (v2damaged + M^-1 * J^T * lambda) + softness * total_lambda + softness * lambda + b = 0
110  //
111  // (J * M^-1 * J^T + softness) * lambda = -(J * v2damaged + softness * total_lambda + b)
112  //
113  // So our lagrange multiplier becomes:
114  //
115  // lambda = -K^-1 (J v + softness * total_lambda + b)
116  //
117  // So we return the bias: softness * total_lambda + b
118  return mSoftness * inTotalLambda + mBias;
119  }
120 
121 private:
122  float mBias;
123  float mSoftness;
124 };
