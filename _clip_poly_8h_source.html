12 template <class VERTEX_ARRAY>
13 void ClipPolyVsPlane(const VERTEX_ARRAY &inPolygonToClip, Vec3Arg inPlaneOrigin, Vec3Arg inPlaneNormal, VERTEX_ARRAY &outClippedPolygon)
14 {
15  JPH_ASSERT(inPolygonToClip.size() >= 2);
16  JPH_ASSERT(outClippedPolygon.empty());
17 
18  // Determine state of last point
19  Vec3 e1 = inPolygonToClip[inPolygonToClip.size() - 1];
20  float prev_num = (inPlaneOrigin - e1).Dot(inPlaneNormal);
21  bool prev_inside = prev_num < 0.0f;
22 
23  // Loop through all vertices
24  for (typename VERTEX_ARRAY::size_type j = 0; j < inPolygonToClip.size(); ++j)
25  {
26  // Check if second point is inside
27  Vec3Arg e2 = inPolygonToClip[j];
28  float num = (inPlaneOrigin - e2).Dot(inPlaneNormal);
29  bool cur_inside = num < 0.0f;
30 
31  // In -> Out or Out -> In: Add point on clipping plane
32  if (cur_inside != prev_inside)
33  {
34  // Solve: (X - inPlaneOrigin) . inPlaneNormal = 0 and X = e1 + t * (e2 - e1) for X
35  Vec3 e12 = e2 - e1;
36  float denom = e12.Dot(inPlaneNormal);
37  outClippedPolygon.push_back(e1 + (prev_num / denom) * e12);
38  }
39 
40  // Point inside, add it
41  if (cur_inside)
42  outClippedPolygon.push_back(e2);
43 
44  // Update previous state
45  prev_num = num;
46  prev_inside = cur_inside;
47  e1 = e2;
48  }
49 }

58 void ClipPolyVsPoly(const VERTEX_ARRAY &inPolygonToClip, const VERTEX_ARRAY &inClippingPolygon, Vec3Arg inClippingPolygonNormal, VERTEX_ARRAY &outClippedPolygon)
59 {
60  JPH_ASSERT(inPolygonToClip.size() >= 2);
61  JPH_ASSERT(inClippingPolygon.size() >= 3);
62 
63  VERTEX_ARRAY tmp_vertices[2];
64  int tmp_vertices_idx = 0;
65 
66  for (typename VERTEX_ARRAY::size_type i = 0; i < inClippingPolygon.size(); ++i)
67  {
68  // Get edge to clip against
69  Vec3 clip_e1 = inClippingPolygon[i];
70  Vec3 clip_e2 = inClippingPolygon[(i + 1) % inClippingPolygon.size()];
71  Vec3 clip_normal = inClippingPolygonNormal.Cross(clip_e2 - clip_e1); // Pointing inward to the clipping polygon
72 
73  // Get source and target polygon
74  const VERTEX_ARRAY &src_polygon = (i == 0)? inPolygonToClip : tmp_vertices[tmp_vertices_idx];
75  tmp_vertices_idx ^= 1;
76  VERTEX_ARRAY &tgt_polygon = (i == inClippingPolygon.size() - 1)? outClippedPolygon : tmp_vertices[tmp_vertices_idx];
77  tgt_polygon.clear();
78 
79  // Clip against the edge
80  ClipPolyVsPlane(src_polygon, clip_e1, clip_normal, tgt_polygon);
81 
82  // Break out if no polygon left
83  if (tgt_polygon.size() < 3)
84  {
85  outClippedPolygon.clear();
86  break;
87  }

93 template <class VERTEX_ARRAY>
94 void ClipPolyVsEdge(const VERTEX_ARRAY &inPolygonToClip, Vec3Arg inEdgeVertex1, Vec3Arg inEdgeVertex2, Vec3Arg inClippingEdgeNormal, VERTEX_ARRAY &outClippedPolygon)
95 {
96  JPH_ASSERT(inPolygonToClip.size() >= 3);
97  JPH_ASSERT(outClippedPolygon.empty());
98 
99  // Get normal that is perpendicular to the edge and the clipping edge normal
100  Vec3 edge = inEdgeVertex2 - inEdgeVertex1;
101  Vec3 edge_normal = inClippingEdgeNormal.Cross(edge);
102 
103  // Project vertices of edge on inPolygonToClip
104  Vec3 polygon_normal = (inPolygonToClip[2] - inPolygonToClip[0]).Cross(inPolygonToClip[1] - inPolygonToClip[0]);
105  float polygon_normal_len_sq = polygon_normal.LengthSq();
106  Vec3 v1 = inEdgeVertex1 + polygon_normal.Dot(inPolygonToClip[0] - inEdgeVertex1) * polygon_normal / polygon_normal_len_sq;
107  Vec3 v2 = inEdgeVertex2 + polygon_normal.Dot(inPolygonToClip[0] - inEdgeVertex2) * polygon_normal / polygon_normal_len_sq;
108  Vec3 v12 = v2 - v1;
109  float v12_len_sq = v12.LengthSq();
110 
111  // Determine state of last point
112  Vec3 e1 = inPolygonToClip[inPolygonToClip.size() - 1];
113  float prev_num = (inEdgeVertex1 - e1).Dot(edge_normal);
114  bool prev_inside = prev_num < 0.0f;
115 
116  // Loop through all vertices
117  for (typename VERTEX_ARRAY::size_type j = 0; j < inPolygonToClip.size(); ++j)
118  {
119  // Check if second point is inside
120  Vec3 e2 = inPolygonToClip[j];
121  float num = (inEdgeVertex1 - e2).Dot(edge_normal);
122  bool cur_inside = num < 0.0f;
123 
124  // In -> Out or Out -> In: Add point on clipping plane
125  if (cur_inside != prev_inside)
126  {
127  // Solve: (X - inPlaneOrigin) . inPlaneNormal = 0 and X = e1 + t * (e2 - e1) for X
128  Vec3 e12 = e2 - e1;
129  float denom = e12.Dot(edge_normal);
130  Vec3 clipped_point = e1 + (prev_num / denom) * e12;
131 
132  // Project point on line segment v1, v2 so see if it falls outside if the edge
133  float projection = (clipped_point - v1).Dot(v12);
134  if (projection < 0.0f)
135  outClippedPolygon.push_back(v1);
136  else if (projection > v12_len_sq)
137  outClippedPolygon.push_back(v2);
138  else
139  outClippedPolygon.push_back(clipped_point);
140  }
141 
142  // Update previous state
143  prev_num = num;
144  prev_inside = cur_inside;
145  e1 = e2;
146  }

152 void ClipPolyVsAABox(const VERTEX_ARRAY &inPolygonToClip, const AABox &inAABox, VERTEX_ARRAY &outClippedPolygon)
153 {
154  JPH_ASSERT(inPolygonToClip.size() >= 2);
155 
156  VERTEX_ARRAY tmp_vertices[2];
157  int tmp_vertices_idx = 0;
158 
159  for (int coord = 0; coord < 3; ++coord)
160  for (int side = 0; side < 2; ++side)
161  {
162  // Get plane to clip against
163  Vec3 origin = Vec3::sZero(), normal = Vec3::sZero();
164  if (side == 0)
165  {
166  normal.SetComponent(coord, 1.0f);
167  origin.SetComponent(coord, inAABox.mMin[coord]);
168  }
169  else
170  {
171  normal.SetComponent(coord, -1.0f);
172  origin.SetComponent(coord, inAABox.mMax[coord]);
173  }
174 
175  // Get source and target polygon
176  const VERTEX_ARRAY &src_polygon = tmp_vertices_idx == 0? inPolygonToClip : tmp_vertices[tmp_vertices_idx & 1];
177  tmp_vertices_idx++;
178  VERTEX_ARRAY &tgt_polygon = tmp_vertices_idx == 6? outClippedPolygon : tmp_vertices[tmp_vertices_idx & 1];
179  tgt_polygon.clear();
180 
181  // Clip against the edge
182  ClipPolyVsPlane(src_polygon, origin, normal, tgt_polygon);
183 
184  // Break out if no polygon left
185  if (tgt_polygon.size() < 3)
186  {
187  outClippedPolygon.clear();
188  return;
189  }
190 
191  // Flip normal
192  normal = -normal;
193  }
194 }
