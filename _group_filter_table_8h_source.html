43  int GetBit(SubGroupID inSubGroup1, SubGroupID inSubGroup2) const
44  {
45  JPH_ASSERT(inSubGroup1 != inSubGroup2);
46 
47  // We store the lower left half only, so swap the inputs when trying to access the top right half
48  if (inSubGroup1 > inSubGroup2)
49  swap(inSubGroup1, inSubGroup2);
50 
51  JPH_ASSERT(inSubGroup2 < mNumSubGroups);
52 
53  // Calculate at which bit the entry for this pair resides
54  // We use the fact that a row always starts at inSubGroup2 * (inSubGroup2 - 1) / 2
55  // (this is the amount of bits needed to store a table of inSubGroup2 entries)
56  return (inSubGroup2 * (inSubGroup2 - 1)) / 2 + inSubGroup1;
57  }

87  virtual bool CanCollide(const CollisionGroup &inGroup1, const CollisionGroup &inGroup2) const override
88  {
89  // If one of the groups is cInvalidGroup the objects will collide (note that the if following this if will ensure that group2 is not cInvalidGroup)
90  if (inGroup1.GetGroupID() == CollisionGroup::cInvalidGroup)
91  return true;
92 
93  // If the objects are in different groups, they collide
94  if (inGroup1.GetGroupID() != inGroup2.GetGroupID())
95  return true;
96 
97  // If the collision filters do not match, but they're in the same group we ignore the collision
98  if (inGroup1.GetGroupFilter() != inGroup2.GetGroupFilter())
99  return false;
100 
101  // If they are in the same sub group, they don't collide
102  if (inGroup1.GetSubGroupID() == inGroup2.GetSubGroupID())
103  return false;
104 
105  // Test if the bit is set for this group pair
106  int bit = GetBit(inGroup1.GetSubGroupID(), inGroup2.GetSubGroupID());
107  return (mTable[bit >> 3] & (1 << (bit & 0b111))) != 0;
108  }
