33  bool GetClosest(float inPrevVLenSq, Vec3 &outV, float &outVLenSq, uint32 &outSet) const
34  {
35 #ifdef JPH_GJK_DEBUG
36  for (int i = 0; i < mNumPoints; ++i)
37  Trace("y[%d] = [%s], |y[%d]| = %g", i, ConvertToString(mY[i]).c_str(), i, mY[i].Length());
38 #endif
39 
40  uint32 set;
41  Vec3 v;
42 
43  switch (mNumPoints)
44  {
45  case 1:
46  // Single point
47  set = 0b0001;
48  v = mY[0];
49  break;
50 
51  case 2:
52  // Line segment
53  v = ClosestPoint::GetClosestPointOnLine(mY[0], mY[1], set);
54  break;
55 
56  case 3:
57  // Triangle
58  v = ClosestPoint::GetClosestPointOnTriangle(mY[0], mY[1], mY[2], set);
59  break;
60 
61  case 4:
62  // Tetrahedron
63  v = ClosestPoint::GetClosestPointOnTetrahedron(mY[0], mY[1], mY[2], mY[3], set);
64  break;
65 
66  default:
67  JPH_ASSERT(false);
68  return false;
69  }
70 
71 #ifdef JPH_GJK_DEBUG
72  Trace("GetClosest: set = 0b%s, v = [%s], |v| = %g", NibbleToBinary(set), ConvertToString(v).c_str(), v.Length());
73 #endif
74 
75  float v_len_sq = v.LengthSq();
76  if (v_len_sq >= inPrevVLenSq)
77  {
78  // No better match found
79 #ifdef JPH_GJK_DEBUG
80  Trace("New closer point is further away, failed to converge");
81 #endif
82  return false;
83  }
84 
85  // Return closest point
86  outV = v;
87  outVLenSq = v_len_sq;
88  outSet = set;
89  return true;
90  }

156  // Calculate closest points on A and B
157  void CalculatePointAAndB(Vec3 &outPointA, Vec3 &outPointB)
158  {
159  switch (mNumPoints)
160  {
161  case 1:
162  outPointA = mP[0];
163  outPointB = mQ[0];
164  break;
165 
166  case 2:
167  {
168  float u, v;
169  ClosestPoint::GetBaryCentricCoordinates(mY[0], mY[1], u, v);
170  outPointA = u * mP[0] + v * mP[1];
171  outPointB = u * mQ[0] + v * mQ[1];

173  break;

179  outPointA = u * mP[0] + v * mP[1] + w * mP[2];
180  outPointB = u * mQ[0] + v * mQ[1] + w * mQ[2];
181  }
182  break;
183 
184  case 4:
185  #ifdef _DEBUG

187  memset(&outPointB, 0xcd, sizeof(outPointB));

203  template <typename A, typename B>
204  bool Intersects(const A &inA, const B &inB, float inTolerance, Vec3 &ioV)
205  {
206  float tolerance_sq = Square(inTolerance);
207 
208  // Reset state
209  mNumPoints = 0;
210 
211 #ifdef JPH_GJK_DEBUG
212  for (int i = 0; i < 4; ++i)
213  mY[i] = Vec3::sZero();
214 #endif
215 
216  // Previous length^2 of v
217  float prev_v_len_sq = FLT_MAX;
218 
219  for (;;)
220  {
221 #ifdef JPH_GJK_DEBUG
222  Trace("v = [%s], num_points = %d", ConvertToString(ioV).c_str(), mNumPoints);
223 #endif
224 
225  // Get support points for shape A and B in the direction of v
226  Vec3 p = inA.GetSupport(ioV);
227  Vec3 q = inB.GetSupport(-ioV);
228 
229  // Get support point of the minkowski sum A - B of v
230  Vec3 w = p - q;
231 
232  // If the support point sA-B(v) is in the opposite direction as v, then we have found a separating axis and there is no intersection
233  if (ioV.Dot(w) < 0.0f)
234  {
235  // Separating axis found
236 #ifdef JPH_GJK_DEBUG
237  Trace("Seperating axis");
238 #endif
239  return false;
240  }
241 
242  // Store the point for later use
243  mY[mNumPoints] = w;
244  ++mNumPoints;
245 
246 #ifdef JPH_GJK_DEBUG
247  Trace("w = [%s]", ConvertToString(w).c_str());
248 #endif
249 
250  // Determine the new closest point
251  float v_len_sq; // Length^2 of v
252  uint32 set; // Set of points that form the new simplex
253  if (!GetClosest(prev_v_len_sq, ioV, v_len_sq, set))
254  return false;
255 
256  // If there are 4 points, the origin is inside the tetrahedron and we're done
257  if (set == 0xf)
258  {
259 #ifdef JPH_GJK_DEBUG
260  Trace("Full simplex");
261 #endif
262  ioV = Vec3::sZero();
263  return true;
264  }
265 
266  // If v is very close to zero, we consider this a collision
267  if (v_len_sq <= tolerance_sq)
268  {
269 #ifdef JPH_GJK_DEBUG
270  Trace("Distance zero");
271 #endif
272  ioV = Vec3::sZero();
273  return true;
274  }
275 
276  // If v is very small compared to the length of y, we also consider this a collision
277  if (v_len_sq <= FLT_EPSILON * GetMaxYLengthSq())
278  {
279 #ifdef JPH_GJK_DEBUG
280  Trace("Machine precision reached");
281 #endif
282  ioV = Vec3::sZero();
283  return true;
284  }
285 
286  // The next seperation axis to test is the negative of the closest point of the Minkowski sum to the origin
287  // Note: This must be done before terminating as converged since the separating axis is -v
288  ioV = -ioV;
289 
290  // If the squared length of v is not changing enough, we've converged and there is no collision
291  JPH_ASSERT(prev_v_len_sq >= v_len_sq);
292  if (prev_v_len_sq - v_len_sq <= FLT_EPSILON * prev_v_len_sq)
293  {
294  // v is a separating axis
295 #ifdef JPH_GJK_DEBUG
296  Trace("Converged");
297 #endif
298  return false;
299  }
300  prev_v_len_sq = v_len_sq;
301 
302  // Update the points of the simplex
303  UpdatePointSetY(set);
304  }
305  }

323  template <typename A, typename B>
324  float GetClosestPoints(const A &inA, const B &inB, float inTolerance, float inMaxDistSq, Vec3 &ioV, Vec3 &outPointA, Vec3 &outPointB)
325  {
326  float tolerance_sq = Square(inTolerance);
327 
328  // Reset state
329  mNumPoints = 0;
330 
331 #ifdef JPH_GJK_DEBUG
332  // Generate the hull of the Minkowski difference for visualization
333  MinkowskiDifference<A, B> diff(inA, inB);
334  mGeometry = DebugRenderer::sInstance->CreateTriangleGeometryForConvex([&diff](Vec3Arg inDirection) { return diff.GetSupport(inDirection); });
335 
336  for (int i = 0; i < 4; ++i)
337  {
338  mY[i] = Vec3::sZero();
339  mP[i] = Vec3::sZero();
340  mQ[i] = Vec3::sZero();
341  }
342 #endif
343 
344  // Length^2 of v
345  float v_len_sq = ioV.LengthSq();
346 
347  // Previous length^2 of v
348  float prev_v_len_sq = FLT_MAX;
349 
350  for (;;)
351  {
352 #ifdef JPH_GJK_DEBUG
353  Trace("v = [%s], num_points = %d", ConvertToString(ioV).c_str(), mNumPoints);
354 #endif
355 
356  // Get support points for shape A and B in the direction of v
357  Vec3 p = inA.GetSupport(ioV);
358  Vec3 q = inB.GetSupport(-ioV);
359 
360  // Get support point of the minkowski sum A - B of v
361  Vec3 w = p - q;
362 
363  float dot = ioV.Dot(w);
364 
365 #ifdef JPH_GJK_DEBUG
366  // Draw -ioV to show the closest point to the origin from the previous simplex
367  DebugRenderer::sInstance->DrawArrow(mOffset, mOffset - ioV, Color::sOrange, 0.05f);
368 
369  // Draw ioV to show where we're probing next
370  DebugRenderer::sInstance->DrawArrow(mOffset, mOffset + ioV, Color::sCyan, 0.05f);
371 
372  // Draw w, the support point
373  DebugRenderer::sInstance->DrawArrow(mOffset, mOffset + w, Color::sGreen, 0.05f);
374  DebugRenderer::sInstance->DrawMarker(mOffset + w, Color::sGreen, 1.0f);
375 
376  // Draw the simplex and the Minkowski difference around it
377  DrawState();
378 #endif
379 
380  // Test if we have a separation of more than inMaxDistSq, in which case we terminate early
381  if (dot < 0.0f && dot * dot > v_len_sq * inMaxDistSq)
382  {
383 #ifdef JPH_GJK_DEBUG
384  Trace("Distance bigger than max");
385 #endif
386 #ifdef _DEBUG
387  memset(&outPointA, 0xcd, sizeof(outPointA));
388  memset(&outPointB, 0xcd, sizeof(outPointB));
389 #endif
390  return FLT_MAX;
391  }
392 
393  // Store the point for later use
394  mY[mNumPoints] = w;
395  mP[mNumPoints] = p;
396  mQ[mNumPoints] = q;
397  ++mNumPoints;
398 
399 #ifdef JPH_GJK_DEBUG
400  Trace("w = [%s]", ConvertToString(w).c_str());
401 #endif
402 
403  uint32 set;
404  if (!GetClosest(prev_v_len_sq, ioV, v_len_sq, set))
405  {
406  --mNumPoints; // Undo add last point
407  break;
408  }
409 
410  // If there are 4 points, the origin is inside the tetrahedron and we're done
411  if (set == 0xf)
412  {
413 #ifdef JPH_GJK_DEBUG
414  Trace("Full simplex");
415 #endif
416  ioV = Vec3::sZero();
417  v_len_sq = 0.0f;
418  break;
419  }
420 
421  // Update the points of the simplex
422  UpdatePointSetYPQ(set);
423 
424  // If v is very close to zero, we consider this a collision
425  if (v_len_sq <= tolerance_sq)
426  {
427 #ifdef JPH_GJK_DEBUG
428  Trace("Distance zero");
429 #endif
430  ioV = Vec3::sZero();
431  v_len_sq = 0.0f;
432  break;
433  }
434 
435  // If v is very small compared to the length of y, we also consider this a collision
436 #ifdef JPH_GJK_DEBUG
437  Trace("Check v small compared to y: %g <= %g", v_len_sq, FLT_EPSILON * GetMaxYLengthSq());
438 #endif
439  if (v_len_sq <= FLT_EPSILON * GetMaxYLengthSq())
440  {
441 #ifdef JPH_GJK_DEBUG
442  Trace("Machine precision reached");
443 #endif
444  ioV = Vec3::sZero();
445  v_len_sq = 0.0f;
446  break;
447  }
448 
449  // The next seperation axis to test is the negative of the closest point of the Minkowski sum to the origin
450  // Note: This must be done before terminating as converged since the separating axis is -v
451  ioV = -ioV;
452 
453  // If the squared length of v is not changing enough, we've converged and there is no collision
454 #ifdef JPH_GJK_DEBUG
455  Trace("Check v not changing enough: %g <= %g", prev_v_len_sq - v_len_sq, FLT_EPSILON * prev_v_len_sq);
456 #endif
457  JPH_ASSERT(prev_v_len_sq >= v_len_sq);
458  if (prev_v_len_sq - v_len_sq <= FLT_EPSILON * prev_v_len_sq)
459  {
460  // v is a separating axis
461 #ifdef JPH_GJK_DEBUG
462  Trace("Converged");
463 #endif
464  break;
465  }
466  prev_v_len_sq = v_len_sq;
467  }
468 
469  // Get the closest points
470  CalculatePointAAndB(outPointA, outPointB);
471 
472 #ifdef JPH_GJK_DEBUG
473  Trace("Return: v = [%s], |v| = %g", ConvertToString(ioV).c_str(), ioV.Length());
474 
475  // Draw -ioV to show the closest point to the origin from the previous simplex
476  DebugRenderer::sInstance->DrawArrow(mOffset, mOffset - ioV, Color::sOrange, 0.05f);
477 
478  // Draw the closest points
479  DebugRenderer::sInstance->DrawMarker(mOffset + outPointA, Color::sGreen, 1.0f);
480  DebugRenderer::sInstance->DrawMarker(mOffset + outPointB, Color::sPurple, 1.0f);
481 
482  // Draw the simplex and the Minkowski difference around it
483  DrawState();
484 #endif
485 
486  JPH_ASSERT(ioV.LengthSq() == v_len_sq);
487  return v_len_sq;
488  }

512  template <typename A>
513  bool CastRay(Vec3Arg inRayOrigin, Vec3Arg inRayDirection, float inTolerance, const A &inA, float &ioLambda)
514  {
515  float tolerance_sq = Square(inTolerance);
516 
517  // Reset state
518  mNumPoints = 0;
519 
520  float lambda = 0.0f;
521  Vec3 x = inRayOrigin;
522  Vec3 v = x - inA.GetSupport(Vec3::sZero());
523  float v_len_sq = FLT_MAX;
524  bool allow_restart = false;
525 
526  for (;;)
527  {
528 #ifdef JPH_GJK_DEBUG
529  Trace("v = [%s], num_points = %d", ConvertToString(v).c_str(), mNumPoints);
530 #endif
531 
532  // Get new support point
533  Vec3 p = inA.GetSupport(v);
534  Vec3 w = x - p;
535 
536 #ifdef JPH_GJK_DEBUG
537  Trace("w = [%s]", ConvertToString(w).c_str());
538 #endif
539 
540  float v_dot_w = v.Dot(w);
541 #ifdef JPH_GJK_DEBUG
542  Trace("v . w = %g", v_dot_w);
543 #endif
544  if (v_dot_w > 0.0f)
545  {
546  // If ray and normal are in the same direction, we've passed A and there's no collision
547  float v_dot_r = v.Dot(inRayDirection);
548 #ifdef JPH_GJK_DEBUG
549  Trace("v . r = %g", v_dot_r);
550 #endif
551  if (v_dot_r >= 0.0f)
552  return false;
553 
554  // Update the lower bound for lambda
555  float delta = v_dot_w / v_dot_r;
556  float old_lambda = lambda;
557  lambda -= delta;
558 #ifdef JPH_GJK_DEBUG
559  Trace("lambda = %g, delta = %g", lambda, delta);
560 #endif
561 
562  // If lambda didn't change, we cannot converge any further and we assume a hit
563  if (old_lambda == lambda)
564  break;
565 
566  // If lambda is bigger or equal than max, we don't have a hit
567  if (lambda >= ioLambda)
568  return false;
569 
570  // Update x to new closest point on the ray
571  x = inRayOrigin + lambda * inRayDirection;
572 
573  // We've shifted x, so reset v_len_sq so that it is not used as early out for GetClosest
574  v_len_sq = FLT_MAX;
575 
576  // We allow rebuilding the simplex once after x changes because the simplex was built
577  // for another x and numerical round off builds up as you keep adding points to an
578  // existing simplex
579  allow_restart = true;
580  }
581 
582  // Add p to set P: P = P U {p}
583  mP[mNumPoints] = p;
584  ++mNumPoints;
585 
586  // Calculate Y = {x} - P
587  for (int i = 0; i < mNumPoints; ++i)
588  mY[i] = x - mP[i];
589 
590  // Determine the new closest point from Y to origin
591  bool needs_restart = false;
592  uint32 set; // Set of points that form the new simplex
593  if (!GetClosest(v_len_sq, v, v_len_sq, set))
594  {
595 #ifdef JPH_GJK_DEBUG
596  Trace("Failed to converge");
597 #endif
598 
599  // We failed to converge, restart
600  needs_restart = true;
601  }
602  else if (set == 0xf)
603  {
604 #ifdef JPH_GJK_DEBUG
605  Trace("Full simplex");
606 #endif
607 
608  // If there are 4 points, x is inside the tetrahedron and we've found a hit
609  // Double check if this is indeed the case
610  if (v_len_sq <= tolerance_sq)
611  break;
612 
613  // We failed to converge, restart
614  needs_restart = true;
615  }
616 
617  if (needs_restart)
618  {
619  // Only allow 1 restart, if we still can't get a closest point
620  // we're so close that we return this as a hit
621  if (!allow_restart)
622  break;
623 
624  // If we fail to converge, we start again with the last point as simplex
625 #ifdef JPH_GJK_DEBUG
626  Trace("Restarting");
627 #endif
628  allow_restart = false;
629  mP[0] = p;
630  mNumPoints = 1;
631  v = x - p;
632  v_len_sq = FLT_MAX;
633  continue;
634  }
635 
636  // Update the points P to form the new simplex
637  // Note: We're not updating Y as Y will shift with x so we have to calculate it every iteration
638  UpdatePointSetP(set);
639 
640  // Check if x is close enough to inA
641  if (v_len_sq <= tolerance_sq)
642  {
643 #ifdef JPH_GJK_DEBUG
644  Trace("Converged");
645 #endif
646  break;
647  }
648  }
649 
650  // Store hit fraction
651  ioLambda = lambda;
652  return true;
653  }

665  template <typename A, typename B>
666  bool CastShape(Mat44Arg inStart, Vec3Arg inDirection, float inTolerance, const A &inA, const B &inB, float &ioLambda)
667  {
668  // Transform the shape to be cast to the starting position
669  using TransformedA = TransformedConvexObject<A>;
670  TransformedA transformed_a(inStart, inA);
671 
672  // Calculate the minkowski difference inB - inA
673  // inA is moving, so we need to add the back side of inB to the front side of inA
674  MinkowskiDifference<B, TransformedA> difference(inB, transformed_a);
675 
676  // Do a raycast against the Minkowski difference
677  return CastRay(Vec3::sZero(), inDirection, inTolerance, difference, ioLambda);
678  }

697  template <typename A, typename B>
698  bool CastShape(Mat44Arg inStart, Vec3Arg inDirection, float inTolerance, const A &inA, const B &inB, float inConvexRadiusA, float inConvexRadiusB, float &ioLambda, Vec3 &outPointA, Vec3 &outPointB, Vec3 &outSeparatingAxis)
699  {
700  float tolerance_sq = Square(inTolerance);
701 
702  // Calculate how close A and B (without their convex radius) need to be to eachother in order for us to consider this a collision
703  float sum_convex_radius = inConvexRadiusA + inConvexRadiusB;
704 
705  // Transform the shape to be cast to the starting position
706  using TransformedA = TransformedConvexObject<A>;
707  TransformedA transformed_a(inStart, inA);
708 
709  // Reset state
710  mNumPoints = 0;
711 
712  float lambda = 0.0f;
713  Vec3 x = Vec3::sZero(); // Since A is already transformed we can start the cast from zero
714  Vec3 v = -inA.GetSupport(Vec3::sZero());
715  float v_len_sq = FLT_MAX;
716  bool allow_restart = false;
717 
718  // Keeps track of separating axis of the previous iteration.
719  // Initialized at zero as we don't know if our first v is actually a separating axis.
720  Vec3 prev_v = Vec3::sZero();
721 
722  for (;;)
723  {
724 #ifdef JPH_GJK_DEBUG
725  Trace("v = [%s], num_points = %d", ConvertToString(v).c_str(), mNumPoints);
726 #endif
727 
728  // Calculate the minkowski difference inB - inA
729  // inA is moving, so we need to add the back side of inB to the front side of inA
730  // Keep the support points on A and B separate so that in the end we can calculate a contact point
731  Vec3 p = transformed_a.GetSupport(-v);
732  Vec3 q = inB.GetSupport(v);
733  Vec3 w = x - (q - p);
734 
735 #ifdef JPH_GJK_DEBUG
736  Trace("w = [%s]", ConvertToString(w).c_str());
737 #endif
738 
739  // Difference from article to this code:
740  // We did not include the convex radius in p and q in order to be able to calculate a good separating axis at the end of the algorithm.
741  // However when moving forward along inDirection we do need to take this into account so that we keep A and B separated by the sum of their convex radii.
742  // From p we have to subtract: inConvexRadiusA * v / |v|
743  // To q we have to add: inConvexRadiusB * v / |v|
744  // This means that to w we have to add: -(inConvexRadiusA + inConvexRadiusB) * v / |v|
745  // So to v . w we have to add: v . (-(inConvexRadiusA + inConvexRadiusB) * v / |v|) = -(inConvexRadiusA + inConvexRadiusB) * |v|
746  float v_dot_w = v.Dot(w) - sum_convex_radius * v.Length();
747 #ifdef JPH_GJK_DEBUG
748  Trace("v . w = %g", v_dot_w);
749 #endif
750  if (v_dot_w > 0.0f)
751  {
752  // If ray and normal are in the same direction, we've passed A and there's no collision
753  float v_dot_r = v.Dot(inDirection);
754 #ifdef JPH_GJK_DEBUG
755  Trace("v . r = %g", v_dot_r);
756 #endif
757  if (v_dot_r >= 0.0f)
758  return false;
759 
760  // Update the lower bound for lambda
761  float delta = v_dot_w / v_dot_r;
762  float old_lambda = lambda;
763  lambda -= delta;
764 #ifdef JPH_GJK_DEBUG
765  Trace("lambda = %g, delta = %g", lambda, delta);
766 #endif
767 
768  // If lambda didn't change, we cannot converge any further and we assume a hit
769  if (old_lambda == lambda)
770  break;
771 
772  // If lambda is bigger or equal than max, we don't have a hit
773  if (lambda >= ioLambda)
774  return false;
775 
776  // Update x to new closest point on the ray
777  x = lambda * inDirection;
778 
779  // We've shifted x, so reset v_len_sq so that it is not used as early out when GetClosest returns false
780  v_len_sq = FLT_MAX;
781 
782  // Now that we've moved, we know that A and B are not intersecting at lambda = 0, so we can update our tolerance to stop iterating
783  // as soon as A and B are inConvexRadiusA + inConvexRadiusB apart
784  tolerance_sq = Square(inTolerance + sum_convex_radius);
785 
786  // We allow rebuilding the simplex once after x changes because the simplex was built
787  // for another x and numerical round off builds up as you keep adding points to an
788  // existing simplex
789  allow_restart = true;
790  }
791 
792  // Add p to set P, q to set Q: P = P U {p}, Q = Q U {q}
793  mP[mNumPoints] = p;
794  mQ[mNumPoints] = q;
795  ++mNumPoints;
796 
797  // Calculate Y = {x} - (Q - P)
798  for (int i = 0; i < mNumPoints; ++i)
799  mY[i] = x - (mQ[i] - mP[i]);
800 
801  // Determine the new closest point from Y to origin
802  bool needs_restart = false;
803  uint32 set; // Set of points that form the new simplex
804  if (!GetClosest(v_len_sq, v, v_len_sq, set))
805  {
806 #ifdef JPH_GJK_DEBUG
807  Trace("Failed to converge");
808 #endif
809 
810  // We failed to converge, restart
811  needs_restart = true;
812  }
813  else if (set == 0xf)
814  {
815 #ifdef JPH_GJK_DEBUG
816  Trace("Full simplex");
817 #endif
818 
819  // If there are 4 points, x is inside the tetrahedron and we've found a hit
820  // Double check that A and B are indeed touching according to our tolerance
821  if (v_len_sq <= tolerance_sq)
822  break;
823 
824  // We failed to converge, restart
825  needs_restart = true;
826  }
827 
828  if (needs_restart)
829  {
830  // Only allow 1 restart, if we still can't get a closest point
831  // we're so close that we return this as a hit
832  if (!allow_restart)
833  break;
834 
835  // If we fail to converge, we start again with the last point as simplex
836 #ifdef JPH_GJK_DEBUG
837  Trace("Restarting");
838 #endif
839  allow_restart = false;
840  mP[0] = p;
841  mQ[0] = q;
842  mNumPoints = 1;
843  v = x - q;
844  v_len_sq = FLT_MAX;
845  continue;
846  }
847 
848  // Update the points P and Q to form the new simplex
849  // Note: We're not updating Y as Y will shift with x so we have to calculate it every iteration
850  UpdatePointSetPQ(set);
851 
852  // Check if A and B are touching according to our tolerance
853  if (v_len_sq <= tolerance_sq)
854  {
855 #ifdef JPH_GJK_DEBUG
856  Trace("Converged");
857 #endif
858  break;
859  }
860 
861  // Store our v to return as separating axis
862  prev_v = v;
863  }
864 
865  // Calculate Y = {x} - (Q - P) again so we can calculate the contact points
866  for (int i = 0; i < mNumPoints; ++i)
867  mY[i] = x - (mQ[i] - mP[i]);
868 
869  // Calculate the offset we need to apply to A and B to correct for the convex radius
870  Vec3 normalized_v = v.NormalizedOr(Vec3::sZero());
871  Vec3 convex_radius_a = inConvexRadiusA * normalized_v;
872  Vec3 convex_radius_b = inConvexRadiusB * normalized_v;
873 
874  // Get the contact point
875  // Note that A and B will coincide when lambda > 0. In this case we calculate only B as it is more accurate as it contains less terms.
876  switch (mNumPoints)
877  {
878  case 1:
879  outPointB = mQ[0] + convex_radius_b;
880  outPointA = lambda > 0.0f? outPointB : mP[0] - convex_radius_a;
881  break;
882 
883  case 2:
884  {
885  float bu, bv;
886  ClosestPoint::GetBaryCentricCoordinates(mY[0], mY[1], bu, bv);
887  outPointB = bu * mQ[0] + bv * mQ[1] + convex_radius_b;
888  outPointA = lambda > 0.0f? outPointB : bu * mP[0] + bv * mP[1] - convex_radius_a;
889  }
890  break;
891 
892  case 3:
893  case 4: // A full simplex, we can't properly determine a contact point! As contact point we take the closest point of the previous iteration.
894  {
895  float bu, bv, bw;
896  ClosestPoint::GetBaryCentricCoordinates(mY[0], mY[1], mY[2], bu, bv, bw);
897  outPointB = bu * mQ[0] + bv * mQ[1] + bw * mQ[2] + convex_radius_b;
898  outPointA = lambda > 0.0f? outPointB : bu * mP[0] + bv * mP[1] + bw * mP[2] - convex_radius_a;
899  }
900  break;
901  }
902 
903  // Store separating axis, in case we have a convex radius we can just return v,
904  // otherwise v will be very small and we resort to returning previous v as an approximation.
905  outSeparatingAxis = sum_convex_radius > 0.0f? -v : -prev_v;
906 
907  // Store hit fraction
908  ioLambda = lambda;
909  return true;
910  }
