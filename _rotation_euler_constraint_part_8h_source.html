39  JPH_INLINE bool ApplyVelocityStep(Body &ioBody1, Body &ioBody2, Vec3Arg inLambda)
40  {
41  // Apply impulse if delta is not zero
42  if (inLambda != Vec3::sZero())
43  {
44  // Calculate velocity change due to constraint
45  //
46  // Impulse:
47  // P = J^T lambda
48  //
49  // Euler velocity integration:
50  // v' = v + M^-1 P
51  if (ioBody1.IsDynamic())
52  ioBody1.GetMotionProperties()->SubAngularVelocityStep(mInvI1.Multiply3x3(inLambda));
53  if (ioBody2.IsDynamic())
54  ioBody2.GetMotionProperties()->AddAngularVelocityStep(mInvI2.Multiply3x3(inLambda));
55  return true;
56  }
57 
58  return false;
59  }

63  static Quat sGetInvInitialOrientation(const Body &inBody1, const Body &inBody2)
64  {
65  // q20 = q10 r0
66  // <=> r0 = q10^-1 q20
67  // <=> r0^-1 = q20^-1 q10
68  //
69  // where:
70  //
71  // q20 = initial orientation of body 2
72  // q10 = initial orientation of body 1
73  // r0 = initial rotation rotation from body 1 to body 2
74  return inBody2.GetRotation().Conjugated() * inBody1.GetRotation();
75  }

109  inline bool SolveVelocityConstraint(Body &ioBody1, Body &ioBody2)
110  {
111  // Calculate lagrange multiplier:
112  //
113  // lambda = -K^-1 (J v + b)
114  Vec3 lambda = mEffectiveMass.Multiply3x3(ioBody1.GetAngularVelocity() - ioBody2.GetAngularVelocity());
115  mTotalLambda += lambda;
116  return ApplyVelocityStep(ioBody1, ioBody2, lambda);

120  inline bool SolvePositionConstraint(Body &ioBody1, Body &ioBody2, QuatArg inInvInitialOrientation, float inBaumgarte)
121  {
122  // Calculate difference in rotation
123  //
124  // The rotation should be:
125  //
126  // q2 = q1 r0
127  //
128  // But because of drift the actual rotation is
129  //
130  // q2 = diff q1 r0
131  // <=> diff = q2 r0^-1 q1^-1
132  //
133  // Where:
134  // q1 = current rotation of body 1
135  // q2 = current rotation of body 2
136  // diff = error that needs to be reduced to zero
137  Quat diff = ioBody2.GetRotation() * inInvInitialOrientation * ioBody1.GetRotation().Conjugated();
138 
139  // A quaternion can be seen as:
140  //
141  // q = [sin(theta / 2) * v, cos(theta/2)]
142  //
143  // Where:
144  // v = rotation vector
145  // theta = rotation angle
146  //
147  // If we assume theta is small (error is small) then sin(x) = x so an approximation of the error angles is:
148  Vec3 error = 2.0f * diff.GetXYZ();
149  if (error != Vec3::sZero())
150  {
151  // Calculate lagrange multiplier (lambda) for Baumgarte stabilization:
152  //
153  // lambda = -K^-1 * beta / dt * C
154  //
155  // We should divide by inDeltaTime, but we should multiply by inDeltaTime in the Euler step below so they're cancelled out
156  Vec3 lambda = -inBaumgarte * mEffectiveMass * error;
157 
158  // Directly integrate velocity change for one time step
159  //
160  // Euler velocity integration:
161  // dv = M^-1 P
162  //
163  // Impulse:
164  // P = J^T lambda
165  //
166  // Euler position integration:
167  // x' = x + dv * dt
168  //
169  // Note we don't accumulate velocities for the stabilization. This is using the approach described in 'Modeling and
170  // Solving Constraints' by Erin Catto presented at GDC 2007. On slide 78 it is suggested to split up the Baumgarte
171  // stabilization for positional drift so that it does not actually add to the momentum. We combine an Euler velocity
172  // integrate + a position integrate and then discard the velocity change.
173  if (ioBody1.IsDynamic())
174  ioBody1.SubRotationStep(mInvI1.Multiply3x3(lambda));
175  if (ioBody2.IsDynamic())
176  ioBody2.AddRotationStep(mInvI2.Multiply3x3(lambda));
177  return true;
178  }
179 
180  return false;
181  }
