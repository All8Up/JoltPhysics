125  Triangle * CreateTriangle(int inIdx0, int inIdx1, int inIdx2, const Vec3 *inPositions)
126  {
127  Triangle *t;
128  if (mNextFree != nullptr)
129  {
130  // Entry available from the free list
131  t = reinterpret_cast<Triangle *>(&mNextFree->mTriangle);
132  mNextFree = mNextFree->mNextFree;
133  }
134  else
135  {
136  // Allocate from never used before triangle store
137  if (mHighWatermark >= cMaxTriangles)
138  return nullptr; // Buffer full
139  t = reinterpret_cast<Triangle *>(&mTriangles[mHighWatermark].mTriangle);
140  ++mHighWatermark;
141  }
142 
143  // Call constructor
144  new (t) Triangle(inIdx0, inIdx1, inIdx2, inPositions);
145 
146  return t;
147  }

279  Triangle * FindFacingTriangle(Vec3Arg inPosition, float &outBestDistSq)
280  {
281  Triangle *best = nullptr;
282  float best_dist_sq = 0.0f;
283 
284  for (Triangle *t : mTriangleQueue)
285  if (!t->mRemoved)
286  {
287  float dot = t->mNormal.Dot(inPosition - t->mCentroid);
288  if (dot > 0.0f)
289  {
290  float dist_sq = dot * dot / t->mNormal.LengthSq();
291  if (dist_sq > best_dist_sq)
292  {
293  best = t;
294  best_dist_sq = dist_sq;
295  }
296  }
297  }
298 
299  outBestDistSq = best_dist_sq;
300  return best;
301  }

304  bool AddPoint(Triangle *inFacingTriangle, int inIdx, float inClosestDistSq, NewTriangles &outTriangles)
305  {
306  // Get position
307  Vec3 pos = mPositions[inIdx];
308 
309 #ifdef JPH_EPA_CONVEX_BUILDER_DRAW
310  // Draw new support point
311  DrawMarker(pos, Color::sYellow, 1.0f);
312 #endif
313 
314 #ifdef JPH_EPA_CONVEX_BUILDER_VALIDATE
315  // Check if structure is intact
316  ValidateTriangles();
317 #endif
318 
319  // Find edge of convex hull of triangles that are not facing the new vertex w
320  Edges edges;
321  if (!FindEdge(inFacingTriangle, pos, edges))
322  return false;
323 
324  // Create new triangles
325  int num_edges = edges.size();
326  for (int i = 0; i < num_edges; ++i)
327  {
328  // Create new triangle
329  Triangle *nt = CreateTriangle(edges[i].mStartIdx, edges[(i + 1) % num_edges].mStartIdx, inIdx);
330  if (nt == nullptr)
331  return false;
332  outTriangles.push_back(nt);
333 
334  // Check if we need to put this triangle in the priority queue
335  if ((nt->mClosestPointInterior && nt->mClosestLenSq < inClosestDistSq) // For the main algorithm
336  || nt->mClosestLenSq < 0.0f) // For when the origin is not inside the hull yet
337  mTriangleQueue.push_back(nt);
338  }
339 
340  // Link edges
341  for (int i = 0; i < num_edges; ++i)
342  {
343  LinkTriangle(outTriangles[i], 0, edges[i].mNeighbourTriangle, edges[i].mNeighbourEdge);
344  LinkTriangle(outTriangles[i], 1, outTriangles[(i + 1) % num_edges], 2);
345  }
346 
347 #ifdef JPH_EPA_CONVEX_BUILDER_VALIDATE
348  // Check if structure is intact
349  ValidateTriangles();
350 #endif
351 
352 #ifdef JPH_EPA_CONVEX_BUILDER_DRAW
353  // Draw state of the hull
354  DrawState();
355 
356  // Increment iteration counter
357  ++mIteration;
358 #endif
359 
360  return true;
361  }

406  void LinkTriangle(Triangle *inT1, int inEdge1, Triangle *inT2, int inEdge2)
407  {
408  JPH_ASSERT(inEdge1 >= 0 && inEdge1 < 3);
409  JPH_ASSERT(inEdge2 >= 0 && inEdge2 < 3);
410  Edge &e1 = inT1->mEdge[inEdge1];
411  Edge &e2 = inT2->mEdge[inEdge2];
412 
413  // Check not connected yet
414  JPH_ASSERT(e1.mNeighbourTriangle == nullptr);
415  JPH_ASSERT(e2.mNeighbourTriangle == nullptr);
416 
417  // Check vertices match
418  JPH_ASSERT(e1.mStartIdx == inT2->GetNextEdge(inEdge2).mStartIdx);
419  JPH_ASSERT(e2.mStartIdx == inT1->GetNextEdge(inEdge1).mStartIdx);
420 
421  // Link up
422  e1.mNeighbourTriangle = inT2;
423  e1.mNeighbourEdge = inEdge2;
424  e2.mNeighbourTriangle = inT1;
425  e2.mNeighbourEdge = inEdge1;
426  }

429  void UnlinkTriangle(Triangle *inT)
430  {
431  // Unlink from neighbours
432  for (int i = 0; i < 3; ++i)
433  {
434  Edge &edge = inT->mEdge[i];
435  if (edge.mNeighbourTriangle != nullptr)
436  {
437  Edge &neighbour_edge = edge.mNeighbourTriangle->mEdge[edge.mNeighbourEdge];
438 
439  // Validate that neighbour points to us
440  JPH_ASSERT(neighbour_edge.mNeighbourTriangle == inT);
441  JPH_ASSERT(neighbour_edge.mNeighbourEdge == i);
442 
443  // Unlink
444  neighbour_edge.mNeighbourTriangle = nullptr;
445  edge.mNeighbourTriangle = nullptr;
446  }
447  }
448 
449  // If this triangle is not in the priority queue, we can delete it now
450  if (!inT->mInQueue)
451  FreeTriangle(inT);
452  }

456  bool FindEdge(Triangle *inFacingTriangle, Vec3Arg inVertex, Edges &outEdges)
457  {
458  // Assert that we were given an empty array
459  JPH_ASSERT(outEdges.empty());
460 
461  // Should start with a facing triangle
462  JPH_ASSERT(inFacingTriangle->IsFacing(inVertex));
463 
464  // Flag as removed
465  inFacingTriangle->mRemoved = true;
466 
467  // Instead of recursing, we build our own stack with the information we need
468  struct StackEntry
469  {
470  Triangle * mTriangle;
471  int mEdge;
472  int mIter;
473  };
474  StackEntry stack[cMaxEdgeLength];
475  int cur_stack_pos = 0;
476 
477  // Start with the triangle / edge provided
478  stack[0].mTriangle = inFacingTriangle;
479  stack[0].mEdge = 0;
480  stack[0].mIter = -1; // Start with edge 0 (is incremented below before use)
481 
482  // Next index that we expect to find, if we don't then there are 'islands'
483  int next_expected_start_idx = -1;
484 
485  for (;;)
486  {
487  StackEntry &cur_entry = stack[cur_stack_pos];
488 
489  // Next iteration
490  if (++cur_entry.mIter >= 3)
491  {
492  // This triangle needs to be removed, unlink it now
493  UnlinkTriangle(cur_entry.mTriangle);
494 
495  // Pop from stack
496  if (--cur_stack_pos < 0)
497  break;
498  }
499  else
500  {
501  // Visit neighbour
502  Edge &e = cur_entry.mTriangle->mEdge[(cur_entry.mEdge + cur_entry.mIter) % 3];
503  Triangle *n = e.mNeighbourTriangle;
504  if (n != nullptr && !n->mRemoved)
505  {
506  // Check if vertex is on the front side of this triangle
507  if (n->IsFacing(inVertex))
508  {
509  // Vertex on front, this triangle needs to be removed
510  n->mRemoved = true;
511 
512  // Add element to the stack of elements to visit
513  cur_stack_pos++;
514  JPH_ASSERT(cur_stack_pos < cMaxEdgeLength);
515  StackEntry &new_entry = stack[cur_stack_pos];
516  new_entry.mTriangle = n;
517  new_entry.mEdge = e.mNeighbourEdge;
518  new_entry.mIter = 0; // Is incremented before use, we don't need to test this edge again since we came from it
519  }
520  else
521  {
522  // Detect if edge doesn't connect to previous edge, if this happens we have found and 'island' which means
523  // the newly added point is so close to the triangles of the hull that we classified some (nearly) coplanar
524  // triangles as before and some behind the point. At this point we just abort adding the point because
525  // we've reached numerical precision.
526  // Note that we do not need to test if the first and last edge connect, since when there are islands
527  // there should be at least 2 disconnects.
528  if (e.mStartIdx != next_expected_start_idx && next_expected_start_idx != -1)
529  return false;
530 
531  // Next expected index is the start index of our neighbour's edge
532  next_expected_start_idx = n->mEdge[e.mNeighbourEdge].mStartIdx;
533 
534  // Vertex behind, keep edge
535  outEdges.push_back(e);
536  }
537  }
538  }
539  }
540 
541  // Assert that we have a fully connected loop
542  JPH_ASSERT(outEdges.empty() || outEdges[0].mStartIdx == next_expected_start_idx);
543 
544 #ifdef JPH_EPA_CONVEX_BUILDER_DRAW
545  // Draw edge of facing triangles
546  for (int i = 0; i < (int)outEdges.size(); ++i)
547  {
548  Vec3 edge_start = cDrawScale * (mPositions[outEdges[i].mStartIdx] + mOffset);
549  DebugRenderer::sInstance->DrawArrow(edge_start, cDrawScale * (mPositions[outEdges[(i + 1) % outEdges.size()].mStartIdx] + mOffset), Color::sYellow, 0.01f);
550  DebugRenderer::sInstance->DrawText3D(edge_start, ConvertToString(outEdges[i].mStartIdx), Color::sWhite);
551  }
552 
553  // Draw the state with the facing triangles removed
554  DrawState();
555 #endif
556 
557  // When we start with two triangles facing away from each other and adding a point that is on the plane,
558  // sometimes we consider the point in front of both causing both triangles to be removed resulting in an empty edge list.
559  // In this case we fail to add the point which will result in no collision reported (the shapes are contacting in 1 point so there's 0 penetration)
560  return outEdges.size() >= 3;
561  }

563 #ifdef JPH_EPA_CONVEX_BUILDER_VALIDATE
565  void ValidateTriangle(const Triangle *inT) const
566  {
567  if (inT->mRemoved)
568  {
569  // Valdiate that removed triangles are not connected to anything
570  for (const Edge &my_edge : inT->mEdge)
571  JPH_ASSERT(my_edge.mNeighbourTriangle == nullptr);
572  }
573  else
574  {
575  for (int i = 0; i < 3; ++i)
576  {
577  const Edge &my_edge = inT->mEdge[i];
578 
579  // Assert that we have a neighbour
580  const Triangle *nb = my_edge.mNeighbourTriangle;
581  JPH_ASSERT(nb != nullptr);
582 
583  if (nb != nullptr)
584  {
585  // Assert that our neighbours edge points to us
586  const Edge &nb_edge = nb->mEdge[my_edge.mNeighbourEdge];
587  JPH_ASSERT(nb_edge.mNeighbourTriangle == inT);
588  JPH_ASSERT(nb_edge.mNeighbourEdge == i);
589 
590  // Assert that the next edge of the neighbour points to the same vertex as this edge's vertex
591  const Edge &nb_next_edge = nb->GetNextEdge(my_edge.mNeighbourEdge);
592  JPH_ASSERT(nb_next_edge.mStartIdx == my_edge.mStartIdx);
593 
594  // Assert that my next edge points to the same vertex as my neighbours vertex
595  const Edge &my_next_edge = inT->GetNextEdge(i);
596  JPH_ASSERT(my_next_edge.mStartIdx == nb_edge.mStartIdx);
597  }
598  }
599  }
600  }

695 EPAConvexHullBuilder::Triangle::Triangle(int inIdx0, int inIdx1, int inIdx2, const Vec3 *inPositions)
696 {
697  // Fill in indexes
698  JPH_ASSERT(inIdx0 != inIdx1 && inIdx0 != inIdx2 && inIdx1 != inIdx2);
699  mEdge[0].mStartIdx = inIdx0;
700  mEdge[1].mStartIdx = inIdx1;
701  mEdge[2].mStartIdx = inIdx2;
702 
703  // Clear links
704  mEdge[0].mNeighbourTriangle = nullptr;
705  mEdge[1].mNeighbourTriangle = nullptr;
706  mEdge[2].mNeighbourTriangle = nullptr;
707 
708  // Get vertex positions
709  Vec3 y0 = inPositions[inIdx0];
710  Vec3 y1 = inPositions[inIdx1];
711  Vec3 y2 = inPositions[inIdx2];
712 
713  // Calculate centroid
714  mCentroid = (y0 + y1 + y2) / 3.0f;
715 
716  // Calculate edges
717  Vec3 y10 = y1 - y0;
718  Vec3 y20 = y2 - y0;
719  Vec3 y21 = y2 - y1;
720 
721  // The most accurate normal is calculated by using the two shortest edges
722  // See: https://box2d.org/posts/2014/01/troublesome-triangle/
723  // The difference in normals is most pronounced when one edge is much smaller than the others (in which case the other 2 must have roughly the same length).
724  // Therefore we can suffice by just picking the shortest from 2 edges and use that with the 3rd edge to calculate the normal.
725  // We first check which of the edges is shorter.
726  float y20_dot_y20 = y20.Dot(y20);
727  float y21_dot_y21 = y21.Dot(y21);
728  if (y20_dot_y20 < y21_dot_y21)
729  {
730  // We select the edges y10 and y20
731  mNormal = y10.Cross(y20);
732 
733  // Check if triangle is degenerate
734  float normal_len_sq = mNormal.LengthSq();
735  if (normal_len_sq > cMinTriangleArea)
736  {
737  // Determine distance between triangle and origin: distance = (centroid - origin) . normal / |normal|
738  // Note that this way of calculating the closest point is much more accurate than first calculating barycentric coordinates and then calculating the closest
739  // point based on those coordinates. Note that we preserve the sign of the distance to check on which side the origin is.
740  float c_dot_n = mCentroid.Dot(mNormal);
741  mClosestLenSq = abs(c_dot_n) * c_dot_n / normal_len_sq;
742 
743  // Calculate closest point to origin using barycentric coordinates:
744  //
745  // v = y0 + l0 * (y1 - y0) + l1 * (y2 - y0)
746  // v . (y1 - y0) = 0
747  // v . (y2 - y0) = 0
748  //
749  // Written in matrix form:
750  //
751  // | y10.y10 y20.y10 | | l0 | = | -y0.y10 |
752  // | y10.y20 y20.y20 | | l1 | | -y0.y20 |
753  //
754  // (y10 = y1 - y0 etc.)
755  //
756  // Cramers rule to invert matrix:
757  float y10_dot_y10 = y10.LengthSq();
758  float y10_dot_y20 = y10.Dot(y20);
759  float determinant = y10_dot_y10 * y20_dot_y20 - y10_dot_y20 * y10_dot_y20;
760  if (determinant > 0.0f) // If determinant == 0 then the system is linearly dependent and the triangle is degenerate, since y10.10 * y20.y20 > y10.y20^2 it should also be > 0
761  {
762  float y0_dot_y10 = y0.Dot(y10);
763  float y0_dot_y20 = y0.Dot(y20);
764  float l0 = (y10_dot_y20 * y0_dot_y20 - y20_dot_y20 * y0_dot_y10) / determinant;
765  float l1 = (y10_dot_y20 * y0_dot_y10 - y10_dot_y10 * y0_dot_y20) / determinant;
766  mLambda[0] = l0;
767  mLambda[1] = l1;
768  mLambdaRelativeTo0 = true;
769 
770  // Check if closest point is interior to the triangle. For a convex hull which contains the origin each face must contain the origin, but because
771  // our faces are triangles, we can have multiple coplanar triangles and only 1 will have the origin as an interior point. We want to use this triangle
772  // to calculate the contact points because it gives the most accurate results, so we will only add these triangles to the priority queue.
773  if (l0 > -cBarycentricEpsilon && l1 > -cBarycentricEpsilon && l0 + l1 < 1.0f + cBarycentricEpsilon)
774  mClosestPointInterior = true;
775  }
776  }
777  }
778  else
779  {
780  // We select the edges y10 and y21
781  mNormal = y10.Cross(y21);
782 
783  // Check if triangle is degenerate
784  float normal_len_sq = mNormal.LengthSq();
785  if (normal_len_sq > cMinTriangleArea)
786  {
787  // Again calculate distance between triangle and origin
788  float c_dot_n = mCentroid.Dot(mNormal);
789  mClosestLenSq = abs(c_dot_n) * c_dot_n / normal_len_sq;
790 
791  // Calculate closest point to origin using barycentric coordinates but this time using y1 as the reference vertex
792  //
793  // v = y1 + l0 * (y0 - y1) + l1 * (y2 - y1)
794  // v . (y0 - y1) = 0
795  // v . (y2 - y1) = 0
796  //
797  // Written in matrix form:
798  //
799  // | y10.y10 -y21.y10 | | l0 | = | y1.y10 |
800  // | -y10.y21 y21.y21 | | l1 | | -y1.y21 |
801  //
802  // Cramers rule to invert matrix:
803  float y10_dot_y10 = y10.LengthSq();
804  float y10_dot_y21 = y10.Dot(y21);
805  float determinant = y10_dot_y10 * y21_dot_y21 - y10_dot_y21 * y10_dot_y21;
806  if (determinant > 0.0f)
807  {
808  float y1_dot_y10 = y1.Dot(y10);
809  float y1_dot_y21 = y1.Dot(y21);
810  float l0 = (y21_dot_y21 * y1_dot_y10 - y10_dot_y21 * y1_dot_y21) / determinant;
811  float l1 = (y10_dot_y21 * y1_dot_y10 - y10_dot_y10 * y1_dot_y21) / determinant;
812  mLambda[0] = l0;
813  mLambda[1] = l1;
814  mLambdaRelativeTo0 = false;
815 
816  // Again check if the closest point is inside the triangle
817  if (l0 > -cBarycentricEpsilon && l1 > -cBarycentricEpsilon && l0 + l1 < 1.0f + cBarycentricEpsilon)
818  mClosestPointInterior = true;
819  }
820  }
821  }
822 }
