43  JPH_INLINE bool ApplyVelocityStep(Body &ioBody1, Body &ioBody2, Vec3Arg inWorldSpaceAxis, float inLambda)
44  {
45  // Apply impulse if delta is not zero
46  if (inLambda != 0.0f)
47  {
48  // Calculate velocity change due to constraint
49  //
50  // Impulse:
51  // P = J^T lambda
52  //
53  // Euler velocity integration:
54  // v' = v + M^-1 P
55  if (ioBody1.IsDynamic())
56  {
57  MotionProperties *mp1 = ioBody1.GetMotionProperties();
58  mp1->SubLinearVelocityStep((inLambda * mp1->GetInverseMass()) * inWorldSpaceAxis);
59  mp1->SubAngularVelocityStep(inLambda * Vec3::sLoadFloat3Unsafe(mInvI1_R1PlusUxAxis));
60  }
61  if (ioBody2.IsDynamic())
62  {
63  MotionProperties *mp2 = ioBody2.GetMotionProperties();
64  mp2->AddLinearVelocityStep((inLambda * mp2->GetInverseMass()) * inWorldSpaceAxis);
65  mp2->AddAngularVelocityStep(inLambda * Vec3::sLoadFloat3Unsafe(mInvI2_R2xAxis));
66  }
67  return true;
68  }
69 
70  return false;
71  }

157  inline bool SolveVelocityConstraint(Body &ioBody1, Body &ioBody2, Vec3Arg inWorldSpaceAxis, float inMinLambda, float inMaxLambda)
158  {
159  // Lagrange multiplier is:
160  //
161  // lambda = -K^-1 (J v + b)
162  float lambda = mEffectiveMass * (inWorldSpaceAxis.Dot(ioBody1.GetLinearVelocity() - ioBody2.GetLinearVelocity()) + Vec3::sLoadFloat3Unsafe(mR1PlusUxAxis).Dot(ioBody1.GetAngularVelocity()) - Vec3::sLoadFloat3Unsafe(mR2xAxis).Dot(ioBody2.GetAngularVelocity()) - mSpringPart.GetBias(mTotalLambda));
163  float new_lambda = Clamp(mTotalLambda + lambda, inMinLambda, inMaxLambda); // Clamp impulse
164  lambda = new_lambda - mTotalLambda; // Lambda potentially got clamped, calculate the new impulse to apply
165  mTotalLambda = new_lambda; // Store accumulated impulse
166 
167  return ApplyVelocityStep(ioBody1, ioBody2, inWorldSpaceAxis, lambda);
168  }

176  inline bool SolvePositionConstraint(Body &ioBody1, Body &ioBody2, Vec3Arg inWorldSpaceAxis, float inC, float inBaumgarte)
177  {
178  // Only apply position constraint when the constraint is hard, otherwise the velocity bias will fix the constraint
179  if (inC != 0.0f && !mSpringPart.IsActive())
180  {
181  // Calculate lagrange multiplier (lambda) for Baumgarte stabilization:
182  //
183  // lambda = -K^-1 * beta / dt * C
184  //
185  // We should divide by inDeltaTime, but we should multiply by inDeltaTime in the Euler step below so they're cancelled out
186  float lambda = -mEffectiveMass * inBaumgarte * inC;
187 
188  // Directly integrate velocity change for one time step
189  //
190  // Euler velocity integration:
191  // dv = M^-1 P
192  //
193  // Impulse:
194  // P = J^T lambda
195  //
196  // Euler position integration:
197  // x' = x + dv * dt
198  //
199  // Note we don't accumulate velocities for the stabilization. This is using the approach described in 'Modeling and
200  // Solving Constraints' by Erin Catto presented at GDC 2007. On slide 78 it is suggested to split up the Baumgarte
201  // stabilization for positional drift so that it does not actually add to the momentum. We combine an Euler velocity
202  // integrate + a position integrate and then discard the velocity change.
203  if (ioBody1.IsDynamic())
204  {
205  ioBody1.SubPositionStep((lambda * ioBody1.GetMotionProperties()->GetInverseMass()) * inWorldSpaceAxis);
206  ioBody1.SubRotationStep(lambda * Vec3::sLoadFloat3Unsafe(mInvI1_R1PlusUxAxis));
207  }
208  if (ioBody2.IsDynamic())
209  {
210  ioBody2.AddPositionStep((lambda * ioBody2.GetMotionProperties()->GetInverseMass()) * inWorldSpaceAxis);
211  ioBody2.AddRotationStep(lambda * Vec3::sLoadFloat3Unsafe(mInvI2_R2xAxis));
212  }
213  return true;
214  }
215 
216  return false;
217  }
