34  const int cMaxSweeps = 50;
35 
36  // Get problem dimension
37  const uint n = inMatrix.GetRows();
38 
39  // Make sure the dimensions are right
40  JPH_ASSERT(inMatrix.GetRows() == n);
41  JPH_ASSERT(inMatrix.GetCols() == n);
42  JPH_ASSERT(outEigVec.GetRows() == n);

44  JPH_ASSERT(outEigVal.GetRows() == n);

48  Matrix a = inMatrix;

51 
52  for (uint ip = 0; ip < n; ++ip)
53  {
54  // Initialize b to diagonal of a
55  b[ip] = a(ip, ip);
56 
57  // Initialize output to diagonal of a
58  outEigVal[ip] = a(ip, ip);
59 
60  // Reset z
61  z[ip] = 0.0f;
62  }
63 
64  for (int sweep = 0; sweep < cMaxSweeps; ++sweep)
65  {
66  // Get the sum of the off-diagonal elements of a
67  float sm = 0.0f;
68  for (uint ip = 0; ip < n - 1; ++ip)
69  for (uint iq = ip + 1; iq < n; ++iq)
70  sm += abs(a(ip, iq));
71 
72  // Normal return, convergence to machine underflow
73  if (sm == 0.0f)
74  {
75  // Sanity checks
76  #ifdef JPH_ENABLE_ASSERTS
77  for (uint c = 0; c < n; ++c)
78  {
79  // Check if the eigenvector is normalized
80  JPH_ASSERT(outEigVec.GetColumn(c).IsNormalized());
81 
82  // Check if inMatrix * eigen_vector = eigen_value * eigen_vector
83  Vector mat_eigvec = inMatrix * outEigVec.GetColumn(c);
84  Vector eigval_eigvec = outEigVal[c] * outEigVec.GetColumn(c);
85  JPH_ASSERT(mat_eigvec.IsClose(eigval_eigvec, max(mat_eigvec.LengthSq(), eigval_eigvec.LengthSq()) * 1.0e-6f));
86  }
87  #endif
88 
89  // Success
90  return true;
91  }
92 
93  // On the first three sweeps use a fraction of the sum of the off diagonal elements as treshold
94  float tresh = sweep < 4? 0.2f * sm / Square(n) : 0.0f;
95 
96  for (uint ip = 0; ip < n - 1; ++ip)
97  for (uint iq = ip + 1; iq < n; ++iq)
98  {
99  float g = 100.0f * abs(a(ip, iq));
100 
101  // After four sweeps, skip the rotation if the off-diagonal element is small
102  if (sweep > 4
103  && abs(outEigVal[ip]) + g == abs(outEigVal[ip])
104  && abs(outEigVal[iq]) + g == abs(outEigVal[iq]))
105  {
106  a(ip, iq) = 0.0f;
107  }
108  else if (abs(a(ip, iq)) > tresh)
109  {
110  float h = outEigVal[iq] - outEigVal[ip];
111 
112  float t;
113  if (abs(h) + g == abs(h))
114  {
115  t = a(ip, iq) / h;
116  }
117  else
118  {
119  float theta = 0.5f * h / a(ip, iq); // Warning: Can become inf if a(ip, iq) too small
120  t = 1.0f / (abs(theta) + sqrt(1.0f + theta * theta)); // Warning: Squaring large value can make it inf
121  if (theta < 0.0f) t = -t;
122  }
123 
124  float c = 1.0f / sqrt(1.0f + t * t);
125  float s = t * c;
126  float tau = s / (1.0f + c);
127  h = t * a(ip, iq);
128 
129  a(ip, iq) = 0.0f;
130 
131  // !Modification from Numerical Recipes!
132  // h can become infinite due to numerical overflow, this only happens when a(ip, iq) is very small
133  // so we can safely set a(ip, iq) to zero and skip the rotation, see lines marked with 'Warning' above.
134  if (!isnan(h))
135  {
136  z[ip] -= h;
137  z[iq] += h;
138 
139  outEigVal[ip] -= h;
140  outEigVal[iq] += h;
141 
142  #define JPH_EVS_ROTATE(a, i, j, k, l) \
143  g = a(i, j), \
144  h = a(k, l), \
145  a(i, j) = g - s * (h + g * tau), \
146  a(k, l) = h + s * (g - h * tau)
147 
148  uint j;
149  for (j = 0; j < ip; ++j) JPH_EVS_ROTATE(a, j, ip, j, iq);
150  for (j = ip + 1; j < iq; ++j) JPH_EVS_ROTATE(a, ip, j, j, iq);
151  for (j = iq + 1; j < n; ++j) JPH_EVS_ROTATE(a, ip, j, iq, j);
152  for (j = 0; j < n; ++j) JPH_EVS_ROTATE(outEigVec, j, ip, j, iq);
153 
154  #undef JPH_EVS_ROTATE
155  }
156  }
157  }
158 
159  // Update eigenvalues with the sum of ta_pq and reinitialize z
160  for (uint ip = 0; ip < n; ++ip)
161  {
162  b[ip] += z[ip];
163  outEigVal[ip] = b[ip];
164  z[ip] = 0.0f;
165  }
166  }
167 
168  // Failure
169  JPH_ASSERT(false, "Too many iterations");
170  return false;
171 }
