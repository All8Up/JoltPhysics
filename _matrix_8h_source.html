231 template <>
232 inline bool Matrix<2, 2>::SetInversed(const Matrix<2, 2> &inM)
233 {
234  // Fetch elements
235  float a = inM.mCol[0].mF32[0];
236  float b = inM.mCol[1].mF32[0];
237  float c = inM.mCol[0].mF32[1];
238  float d = inM.mCol[1].mF32[1];
239 
240  // Calculate determinant
241  float det = a * d - b * c;
242  if (det == 0.0f)
243  return false;
244 
245  // Construct inverse
246  mCol[0].mF32[0] = d / det;
247  mCol[1].mF32[0] = -b / det;
248  mCol[0].mF32[1] = -c / det;
249  mCol[1].mF32[1] = a / det;
250  return true;
