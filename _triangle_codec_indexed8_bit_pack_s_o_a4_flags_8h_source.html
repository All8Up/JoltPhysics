109  uint Pack(const VertexList &inVertices, const IndexedTriangleList &inTriangles, Vec3Arg inBoundsMin, Vec3Arg inBoundsMax, ByteBuffer &ioBuffer, string &outError)
110  {
111  // Determine position of triangles start
112  uint offset = (uint)ioBuffer.size();
113 
114  // Update stats
115  uint tri_count = (uint)inTriangles.size();
116  mNumTriangles += tri_count;
117 
118  // Allocate triangle block header
119  TriangleBlockHeader *header = ioBuffer.Allocate<TriangleBlockHeader>();
120 
121  // Compute first vertex that this batch will use (ensuring there's enough room if none of the vertices are shared)
122  uint start_vertex = Clamp((int)mVertices.size() - 256 + (int)tri_count * 3, 0, (int)mVertices.size());
123 
124  // Store the start vertex offset, this will later be patched to give the delta offset relative to the triangle block
125  mOffsetsToPatch.push_back(uint((uint8 *)&header->mOffsetToVertices - &ioBuffer[0]));
126  header->mOffsetToVertices = start_vertex * sizeof(VertexData);
127 
128  // Pack vertices
129  uint padded_triangle_count = AlignUp(tri_count, 4);
130  for (uint t = 0; t < padded_triangle_count; t += 4)
131  {
132  TriangleBlock *block = ioBuffer.Allocate<TriangleBlock>();
133  for (uint vertex_nr = 0; vertex_nr < 3; ++vertex_nr)
134  for (uint block_tri_idx = 0; block_tri_idx < 4; ++block_tri_idx)
135  {
136  // Fetch vertex index. Create degenerate triangles for padding triangles.
137  bool triangle_available = t + block_tri_idx < tri_count;
138  uint32 src_vertex_index = triangle_available? inTriangles[t + block_tri_idx].mIdx[vertex_nr] : inTriangles[tri_count - 1].mIdx[0];
139 
140  // Check if we've seen this vertex before and if it is in the range that we can encode
141  uint32 vertex_index;
142  VertexMap::const_iterator found = mVertexMap.find(src_vertex_index);
143  if (found == mVertexMap.end() || found->second < start_vertex)
144  {
145  // Add vertex
146  vertex_index = (uint32)mVertices.size();
147  mVertexMap[src_vertex_index] = vertex_index;
148  mVertices.push_back(inVertices[src_vertex_index]);
149  }
150  else
151  {
152  // Reuse vertex
153  vertex_index = found->second;
154  }
155 
156  // Store vertex index
157  uint32 vertex_offset = vertex_index - start_vertex;
158  if (vertex_offset > 0xff)
159  {
160  outError = "TriangleCodecIndexed8BitPackSOA4Flags: Offset doesn't fit in 8 bit";
161  return uint(-1);
162  }
163  block->mIndices[vertex_nr][block_tri_idx] = (uint8)vertex_offset;
164 
165  // Store flags
166  uint32 flags = triangle_available? inTriangles[t + block_tri_idx].mMaterialIndex : 0;
167  if (flags > 0xff)
168  {
169  outError = "TriangleCodecIndexed8BitPackSOA4Flags: Material index doesn't fit in 8 bit";
170  return uint(-1);
171  }
172  block->mFlags[block_tri_idx] = (uint8)flags;
173  }
174  }
175 
176  return offset;
177  }

180  void Finalize(TriangleHeader *ioHeader, ByteBuffer &ioBuffer) const
181  {
182  // Check if anything to do
183  if (mVertices.empty())
184  return;
185 
186  // Align buffer to 4 bytes
187  uint vertices_idx = (uint)ioBuffer.Align(4);
188 
189  // Patch the offsets
190  for (uint o : mOffsetsToPatch)
191  *ioBuffer.Get<uint32>(o) += vertices_idx - o;
192 
193  // Calculate bounding box
194  AABox bounds;
195  for (const Float3 &v : mVertices)
196  bounds.Encapsulate(Vec3(v));
197 
198  // Compress vertices
199  VertexData *vertices = ioBuffer.Allocate<VertexData>(mVertices.size());
200  Vec3 compress_scale = Vec3::sReplicate(COMPONENT_MASK) / Vec3::sMax(bounds.GetSize(), Vec3::sReplicate(1.0e-20f));
201  for (const Float3 &v : mVertices)
202  {
203  UVec4 c = ((Vec3(v) - bounds.mMin) * compress_scale + Vec3::sReplicate(0.5f)).ToInt();
204  JPH_ASSERT(c.GetX() <= COMPONENT_MASK);
205  JPH_ASSERT(c.GetY() <= COMPONENT_MASK);
206  JPH_ASSERT(c.GetZ() <= COMPONENT_MASK);
207  vertices->mVertexXY = c.GetX() + (c.GetY() << COMPONENT_Y1);
208  vertices->mVertexZY = c.GetZ() + ((c.GetY() >> COMPONENT_Y1_BITS) << COMPONENT_Y2);
209  ++vertices;
210  }
211 
212  // Store decompression information
213  bounds.mMin.StoreFloat3(&ioHeader->mOffset);
214  (bounds.GetSize() / Vec3::sReplicate(COMPONENT_MASK)).StoreFloat3(&ioHeader->mScale);
215  }

267  JPH_INLINE void Unpack(Vec3Arg inBoundsMin, Vec3Arg inBoundsMax, const void *inTriangleStart, uint32 inNumTriangles, Vec3 *outTriangles) const
268  {
269  JPH_ASSERT(inNumTriangles > 0);
270  const TriangleBlockHeader *header = reinterpret_cast<const TriangleBlockHeader *>(inTriangleStart);
271  const VertexData *vertices = header->GetVertexData();
272  const TriangleBlock *t = header->GetTriangleBlock();
273  const TriangleBlock *end = t + ((inNumTriangles + 3) >> 2);
274 
275  int triangles_left = inNumTriangles;
276 
277  do
278  {
279  // Get the indices for the three vertices (reads 4 bytes extra, but these are the flags so that's ok)
280  UVec4 indices = UVec4::sLoadInt4(reinterpret_cast<const uint32 *>(&t->mIndices[0]));
281  UVec4 iv1 = indices.Expand4Byte0();
282  UVec4 iv2 = indices.Expand4Byte4();
283  UVec4 iv3 = indices.Expand4Byte8();
284 
285  // Decompress the triangle data
286  Vec4 v1x, v1y, v1z, v2x, v2y, v2z, v3x, v3y, v3z;
287  Unpack(vertices, iv1, v1x, v1y, v1z);
288  Unpack(vertices, iv2, v2x, v2y, v2z);
289  Unpack(vertices, iv3, v3x, v3y, v3z);
290 
291  // Transpose it so we get normal vectors
292  Mat44 v1 = Mat44(v1x, v1y, v1z, Vec4::sZero()).Transposed();
293  Mat44 v2 = Mat44(v2x, v2y, v2z, Vec4::sZero()).Transposed();
294  Mat44 v3 = Mat44(v3x, v3y, v3z, Vec4::sZero()).Transposed();
295 
296  // Store triangle data
297  for (int i = 0; i < 4 && triangles_left > 0; ++i, --triangles_left)
298  {
299  *outTriangles++ = v1.GetColumn3(i);
300  *outTriangles++ = v2.GetColumn3(i);
301  *outTriangles++ = v3.GetColumn3(i);
302  }
303 
304  ++t;
305  }
306  while (t < end);
307  }

310  JPH_INLINE float TestRay(Vec3Arg inRayOrigin, Vec3Arg inRayDirection, Vec3Arg inBoundsMin, Vec3Arg inBoundsMax, const void *inTriangleStart, uint32 inNumTriangles, float inClosest, uint32 &outClosestTriangleIndex) const

314  const VertexData *vertices = header->GetVertexData();
315  const TriangleBlock *t = header->GetTriangleBlock();
316  const TriangleBlock *end = t + ((inNumTriangles + 3) >> 2);
317

319  UVec4 closest_triangle_idx = UVec4::sZero();
320 
321  UVec4 start_triangle_idx = UVec4::sZero();
322  do
323  {
324  // Get the indices for the three vertices (reads 4 bytes extra, but these are the flags so that's ok)

326  UVec4 iv1 = indices.Expand4Byte0();
327  UVec4 iv2 = indices.Expand4Byte4();
328  UVec4 iv3 = indices.Expand4Byte8();
329 
330  // Decompress the triangle data
331  Vec4 v1x, v1y, v1z, v2x, v2y, v2z, v3x, v3y, v3z;
332  Unpack(vertices, iv1, v1x, v1y, v1z);
333  Unpack(vertices, iv2, v2x, v2y, v2z);
334  Unpack(vertices, iv3, v3x, v3y, v3z);
335 
336  // Perform ray vs triangle test
337  Vec4 distance = RayTriangle4(inRayOrigin, inRayDirection, v1x, v1y, v1z, v2x, v2y, v2z, v3x, v3y, v3z);
338 
339  // Update closest with the smaller values
340  UVec4 smaller = Vec4::sLess(distance, closest);
341  closest = Vec4::sSelect(closest, distance, smaller);
342 
343  // Update triangle index with the smallest values
344  UVec4 triangle_idx = start_triangle_idx + UVec4(0, 1, 2, 3);
345  closest_triangle_idx = UVec4::sSelect(closest_triangle_idx, triangle_idx, smaller);
346 
347  // Next block
348  ++t;
349  start_triangle_idx += UVec4::sReplicate(4);
350  }
351  while (t < end);
352 
353  // Get the smallest component
354  Vec4::sSort4(closest, closest_triangle_idx);
355  outClosestTriangleIndex = closest_triangle_idx.GetX();
356  return closest.GetX();
357  }

360  inline void GetTriangle(const void *inTriangleStart, uint32 inTriangleIdx, Vec3 &outV1, Vec3 &outV2, Vec3 &outV3) const
361  {
362  const TriangleBlockHeader *header = reinterpret_cast<const TriangleBlockHeader *>(inTriangleStart);
363  const VertexData *vertices = header->GetVertexData();
364  const TriangleBlock *block = header->GetTriangleBlock() + (inTriangleIdx >> 2);
365  uint32 block_triangle_idx = inTriangleIdx & 0b11;
366 
367  // Get the 3 vertices
368  const VertexData &v1 = vertices[block->mIndices[0][block_triangle_idx]];
369  const VertexData &v2 = vertices[block->mIndices[1][block_triangle_idx]];
370  const VertexData &v3 = vertices[block->mIndices[2][block_triangle_idx]];
371 
372  // Pack the vertices
373  UVec4 c1(v1.mVertexXY, v2.mVertexXY, v3.mVertexXY, 0);
374  UVec4 c2(v1.mVertexZY, v2.mVertexZY, v3.mVertexZY, 0);
375 
376  // Unpack the x y and z component
377  UVec4 xc = UVec4::sAnd(c1, UVec4::sReplicate(COMPONENT_MASK));
378  UVec4 yc = UVec4::sOr(c1.LogicalShiftRight<COMPONENT_Y1>(), c2.LogicalShiftRight<COMPONENT_Y2>().LogicalShiftLeft<COMPONENT_Y1_BITS>());
379  UVec4 zc = UVec4::sAnd(c2, UVec4::sReplicate(COMPONENT_MASK));
380 
381  // Convert to float
382  Vec4 vx = Vec4::sFusedMultiplyAdd(xc.ToFloat(), mScaleX, mOffsetX);
383  Vec4 vy = Vec4::sFusedMultiplyAdd(yc.ToFloat(), mScaleY, mOffsetY);
384  Vec4 vz = Vec4::sFusedMultiplyAdd(zc.ToFloat(), mScaleZ, mOffsetZ);
385 
386  // Transpose it so we get normal vectors
387  Mat44 trans = Mat44(vx, vy, vz, Vec4::sZero()).Transposed();
388  outV1 = trans.GetAxisX();
389  outV2 = trans.GetAxisY();
390  outV3 = trans.GetAxisZ();
391  }
